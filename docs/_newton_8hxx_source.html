58 bool solve(Vector const &x_ini, Vector &x_sol) override
59 {
60 // Reset internal variables
61 this->reset();
62
63 // Initialize variables
64 Real residuals, step_norm;
65 Vector x_old, x_new, function, step;
66 Matrix jacobian;
67
68 // Set initial iteration
69 x_old = x_ini;
70 this->evaluate_function(x_old, function);
71 this->evaluate_jacobian(x_old, jacobian);
72
73 // Algorithm iterations
74 Real tolerance_residuals{this->m_tolerance};
75 Real tolerance_step_norm{this->m_tolerance * this->m_tolerance};
76 for (this->m_iterations = Size(1); this->m_iterations < this->m_max_iterations; ++this->m_iterations)
77 {
78
79 // Calculate step
80 this->m_lu.compute(jacobian);
81 SANDALS_ASSERT(this->m_lu.rank() == N, "Sandals:Newton::solve(...): singular Jacobian detected.");
82 step = -this->m_lu.solve(function);
83
84 // Check convergence
85 residuals = function.norm();
86 step_norm = step.norm();
87 if (residuals < tolerance_residuals || step_norm < tolerance_step_norm) {
88 this->m_converged = true;
89 break;
90 }
91
92 // Update point
93 x_new = x_old + step;
94 this->evaluate_function(x_new, function);
95 this->evaluate_jacobian(x_new, jacobian);
96
97 // Update internal variables
98 x_old = x_new;
99 }
100
101 // Convergence data
102 x_sol = x_old;
103 this->m_residuals = residuals;
104 return this->m_converged;

110 bool solve_damped(Vector const &x_ini, Vector &x_sol) override
111 {
112 // Setup internal variables
113 this->reset();
114
115 // Initialize variables
116 Real residuals_old, residuals_new, step_norm_old, step_norm_new, tau;
117 Vector x_old, x_new, function_old, function_new, step_old, step_new;
118 Matrix jacobian;
119
120 // Set initial iteration
121 x_old = x_ini;
122 this->evaluate_function(x_old, function_old);
123 this->evaluate_jacobian(x_old, jacobian);
124
125 // Algorithm iterations
126 Real tolerance_residuals{this->m_tolerance};
127 Real tolerance_step_norm{this->m_tolerance * this->m_tolerance};
128 for (this->m_iterations = Size(1); this->m_iterations < this->m_max_iterations; ++this->m_iterations)
129 {
130
131 // Calculate step
132 this->m_lu.compute(jacobian);
133 SANDALS_ASSERT(this->m_lu.rank() == N, "Sandals:Newton::solve_damped(...): singular Jacobian detected.");
134 step_old = -this->m_lu.solve(function_old);
135
136 // Check convergence
137 residuals_old = function_old.norm();
138 step_norm_old = step_old.norm();
139 if (residuals_old < tolerance_residuals || step_norm_old < tolerance_step_norm) {
140 this->m_converged = true;
141 break;
142 }
143
144 // Relax the iteration process
145 tau = Real(1.0);
146 for (this->m_relaxations = Size(0); this->m_relaxations < this->m_max_relaxations; ++this->m_relaxations)
147 {
148 // Update point
149 step_new = tau * step_old;
150 x_new = x_old + step_new;
151 this->evaluate_function(x_new, function_new);
152
153 // Check relaxation
154 residuals_new = function_new.norm();
155 step_norm_new = step_new.norm();
156 if (residuals_new < residuals_old || step_norm_new < (Real(1.0)-tau/Real(2.0))*step_norm_old) {
157 this->evaluate_jacobian(x_new, jacobian);
158 break;
159 } else {
160 tau *= this->m_alpha;
161 }
162 }
163
164 // Update internal variables
165 x_old = x_new;
166 function_old = function_new;
167 step_old = step_new;
168 }
169
170 // Convergence data
171 x_sol = x_old;
172 this->m_residuals = residuals_old;
173 return this->m_converged;
174 }

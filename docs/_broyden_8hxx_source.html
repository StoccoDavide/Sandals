78 bool solve(Vector const &x_ini, Vector &x_sol) override
79 {
80 // Setup internal variables
81 this->reset();
82
83 // Initialize variables
84 Real residuals, step_norm;
85 Vector x_old, x_new, function_old, function_new, step_old, step_new, delta_x_old, delta_x_new,
86 delta_function_old, delta_function_new;
87 Matrix jacobian_old, jacobian_new;
88
89 // Set initial iteration
90 x_old = x_ini;
91 this->evaluate_function(x_old, function_old);
92 this->evaluate_jacobian(x_old, jacobian_old);
93
94 // Algorithm iterations
95 Real tolerance_residuals{this->m_tolerance};
96 Real tolerance_step_norm{this->m_tolerance * this->m_tolerance};
97 for (this->m_iterations = Size(1); this->m_iterations < this->m_max_iterations; ++this->m_iterations)
98 {
99
100 // Calculate step
101 step_old = -jacobian_old * function_old;
102
103 // Check convergence
104 residuals = function_old.norm();
105 step_norm = step_old.norm();
106 if (residuals < tolerance_residuals || step_norm < tolerance_step_norm) {
107 this->m_converged = true;
108 break;
109 }
110
111 // Update point
112 x_new = x_old + step_old;
113 this->evaluate_function(x_new, function_new);
114
115 // Update jacobian approximation
116 delta_x_new = x_new - x_old;
117 delta_function_new = function_new - function_old;
118 this->update(
119 delta_x_old, delta_function_old, jacobian_old, // Old step data
120 delta_x_new, delta_function_new, jacobian_new // New step data
121 );
122
123 // Update internal variables
124 x_old = x_new;
125 function_old = function_new;
126 delta_x_old = delta_x_new;
127 delta_function_old = delta_function_new;
128 step_old = step_new;
129 jacobian_old = jacobian_new;
130 }
131
132 // Convergence data
133 x_sol = x_old;
134 this->m_residuals = residuals;
135 return this->m_converged;
136 }

142 bool solve_damped(Vector const &x_ini, Vector &x_sol) override
143 {
144 // Setup internal variables
145 this->reset();
146
147 // Initialize variables
148 Real residuals_old, residuals_new, step_norm_old, step_norm_new, tau;
149 Vector x_old, x_new, function_old, function_new, step_old, step_new, delta_x_old, delta_x_new,
150 delta_function_old, delta_function_new;
151 Matrix jacobian_old, jacobian_new;
152
153 // Set initial iteration
154 x_old = x_ini;
155 this->evaluate_function(x_old, function_old);
156 this->evaluate_jacobian(x_old, jacobian_old);
157
158 // Algorithm iterations
159 Real tolerance_residuals{this->m_tolerance};
160 Real tolerance_step_norm{this->m_tolerance * this->m_tolerance};
161 for (this->m_iterations = Size(1); this->m_iterations < this->m_max_iterations; ++this->m_iterations)
162 {
163
164 // Calculate step
165 step_old = -jacobian_old * function_old;
166
167 // Check convergence
168 residuals_old = function_old.norm();
169 step_norm_old = step_old.norm();
170 if (residuals_old < tolerance_residuals || step_norm_old < tolerance_step_norm) {
171 this->m_converged = true;
172 break;
173 }
174
175 // Relax the iteration process
176 tau = Real(1.0);
177 for (this->m_relaxations = Size(0); this->m_relaxations < this->m_max_relaxations; ++this->m_relaxations)
178 {
179 // Update point
180 step_new = tau * step_old;
181 x_new = x_old + step_new;
182 this->evaluate_function(x_new, function_new);
183
184 // Check relaxation
185 residuals_new = function_new.norm();
186 step_norm_new = step_new.norm();
187 if (residuals_new < residuals_old || step_norm_new < (Real(1.0)-tau/Real(2.0))*step_norm_old) {
188 break;
189 } else {
190 tau *= this->m_alpha;
191 }
192 }
193
194 // Update jacobian approximation
195 delta_x_new = x_new - x_old;
196 delta_function_new = function_new - function_old;
197 this->update(
198 delta_x_old, delta_function_old, jacobian_old, // Old step data
199 delta_x_new, delta_function_new, jacobian_new // New step data
200 );
201
202 // Update internal variables
203 x_old = x_new;
204 function_old = function_new;
205 delta_x_old = delta_x_new;
206 delta_function_old = delta_function_new;
207 step_old = step_new;
208 jacobian_old = jacobian_new;
209 }
210
211 // Convergence data
212 x_sol = x_old;
213 this->m_residuals = residuals_old;
214 return this->m_converged;
215 }

224 void update(
225 Vector const &delta_x_old, Vector const &delta_function_old, Matrix const &jacobian_old,
226 Vector const &delta_x_new, Vector const &delta_function_new, Matrix &jacobian_new
227 ) {
228 Vector tmp_1(jacobian_old * delta_function_new);
229 Real tmp_2{delta_function_new.squaredNorm()};
230 // Selection criteria
231 // |(dx_new'*dx_old) / (dx_new'*J_old*dF_new)| < |(dF_new'*dF_old) / (dF_new'*dF_new)|
232 if (this->m_type == Type::COMBINED || this->m_type == Type::GOOD || this->iterations() < Size(2) ||
233 std::abs(delta_x_new.transpose() * delta_x_old) / std::abs(delta_x_new.transpose() * tmp_1)
234 < std::abs(delta_function_new.transpose() * delta_function_old) / tmp_2) {
235 // Broyden's Good solver
236 // J_new = J_old - (J_old*dF_new-dx_new) / (C'*dF_new)*C', where C = J_old'*dx_new;
237 Vector C_g(jacobian_old.transpose() * delta_x_new);
238 jacobian_new = jacobian_old - (tmp_1 - delta_x_new) / (C_g.transpose() * delta_function_new) * C_g.transpose();
239 } else {
240 // Broyden's Bad solver
241 // J_new = J_old - (J_old*dF_old-dx_new) / (C'*dF_old)*C', where C = dF_old;
242 jacobian_new = jacobian_old - (tmp_1 - delta_x_new) / tmp_2 * delta_function_old.transpose();
243 }
244 }

624 Real estimate_step(VectorN const &x, VectorN const &x_e, Real h_k) const
625 {
626 Real desired_error{this->m_absolute_tolerance + this->m_relative_tolerance *
627 std::max(x.array().abs().maxCoeff(), x_e.array().abs().maxCoeff())};
628 Real truncation_error{(x - x_e).array().abs().maxCoeff()};
629 return h_k * std::min(this->m_max_safety_factor, std::max(this->m_min_safety_factor,
630 this->m_safety_factor * std::pow(desired_error/truncation_error,
631 1.0/std::max(this->m_tableau.order, this->m_tableau.order_e))));
632 }

689 bool erk_explicit_step(VectorN const &x_old, Real t_old, Real h_old, VectorN &x_new, Real &h_new) const
690 {
691 using Eigen::all;
692 using Eigen::seqN;
693
694 // Compute the K variables in the case of an explicit method and explicit system
695 VectorN x_node;
696 MatrixK K;
697 for (Integer i{0}; i < S; ++i) {
698 x_node = x_old + K(all, seqN(0, i)) * this->m_tableau.A(i, seqN(0, i)).transpose();
699 if (!this->m_reverse) {
700 K.col(i) = h_old * static_cast<Explicit<Real, N, M> const *>(this->m_system.get())->f(x_node, t_old + h_old*this->m_tableau.c(i));
701 } else {
702 K.col(i) = h_old * static_cast<Explicit<Real, N, M> const *>(this->m_system.get())->f_reverse(x_node, t_old + h_old*this->m_tableau.c(i));
703 }
704 }
705 if (!K.allFinite()) {return false;}
706
707 // Perform the step and obtain the next state
708 x_new = x_old + K * this->m_tableau.b;
709
710 // Adapt next step
711 if (this->m_adaptive && this->m_tableau.is_embedded) {
712 VectorN x_emb = x_old + K * this->m_tableau.b_e;
713 h_new = this->estimate_step(x_new, x_emb, h_old);
714 }
715 return true;
716 }

922 void irk_jacobian(VectorN const &x, Real t, Real h, VectorK const &K, MatrixJ &jac) const
923 {
924 using Eigen::seqN;
925
926 // Reset the Jacobian matrix
927 jac.setZero();
928
929 // Loop through each equation of the system
930 MatrixK K_mat{K.reshaped(N, S)};
931 Real t_node;
932 VectorN x_node, x_dot_node;
933 MatrixN JF_x, JF_x_dot;
934 auto idx = seqN(0, N), jdx = seqN(0, N);
935 for (Integer i{0}; i < S; ++i) {
936 t_node = t + h * this->m_tableau.c(i);
937 x_node = x + K_mat * this->m_tableau.A.row(i).transpose();
938
939 // Compute the Jacobians with respect to x and x_dot
940 x_dot_node = K_mat.col(i) / h;
941 if (!this->m_reverse) {
942 JF_x = this->m_system->JF_x(x_node, x_dot_node, t_node);
943 JF_x_dot = this->m_system->JF_x_dot(x_node, x_dot_node, t_node);
944 } else {
945 JF_x = this->m_system->JF_x_reverse(x_node, x_dot_node, t_node);
946 JF_x_dot = this->m_system->JF_x_dot_reverse(x_node, x_dot_node, t_node);
947 }
948
949 // Combine the Jacobians with respect to x and x_dot to obtain the Jacobian with respect to K
950 idx = seqN(i*N, N);
951 for (Integer j{0}; j < S; ++j) {
952 jdx = seqN(j*N, N);
953 if (i == j) {
954 jac(idx, jdx) = this->m_tableau.A(i,j) * JF_x + JF_x_dot / h;
955 } else {
956 jac(idx, jdx) = this->m_tableau.A(i,j) * JF_x;
957 }
958 }
959 }
960 }

1178 bool advance(VectorN const &x_old, Real t_old, Real h_old, VectorN &x_new, Real &h_new)
1179 {
1180 #define CMD "Sandals::RungeKutta::advance(...): "
1181
1182 // Check step size
1183 SANDALS_ASSERT(h_old > Real(0.0), CMD "in " << this->m_tableau.name << " solver, h = "<<
1184 h_old << ", expected > 0.");
1185
1186 // If the integration step failed, try again with substepping
1187 if (!this->step(x_old, t_old, h_old, x_new, h_new))
1188 {
1189 VectorN x_tmp(x_old);
1190 Real t_tmp{t_old}, h_tmp{h_old / Real(2.0)};
1191
1192 // Substepping logic
1193 Integer max_k{this->m_max_substeps * this->m_max_substeps}, k{2};
1194 Real h_new_tmp;
1195 while (k > 0) {
1196 // Calculate the next step with substepping logic
1197 if (this->step(x_tmp, t_tmp, h_tmp, x_new, h_new_tmp)) {
1198
1199 // Accept the step
1200 h_tmp = h_new_tmp;
1201
1202 // If substepping is enabled, double the step size
1203 if (k > 0 && k < max_k) {
1204 k -= 1;
1205 // If the substepping index is even, double the step size
1206 if (k % 2 == 0) {
1207 h_tmp = Real(2.0) * h_tmp;
1208 if (this->m_verbose) {
1209 SANDALS_WARNING(CMD "in " << this->m_tableau.name << " solver, at t = " << t_tmp <<
1210 ", integration succedded disable one substepping layer.");
1211 }
1212 }
1213 }
1214
1215 // Check the infinity norm of the solution
1216 SANDALS_ASSERT(std::isfinite(x_tmp.maxCoeff()), CMD "in " << this->m_tableau.name <<
1217 " solver, at t = " << t_tmp << ", ||x||_inf = inf, computation interrupted.");
1218
1219 } else {
1220
1221 // If the substepping index is too high, abort the integration
1222 k += 2;
1223 SANDALS_ASSERT(k < max_k, CMD "in " << this->m_tableau.name << " solver, at t = " <<
1224 t_tmp << ", integration failed with h = " << h_tmp << ", aborting.");
1225 return false;
1226
1227 // Otherwise, try again with a smaller step
1228 if (this->m_verbose) {SANDALS_WARNING(CMD "in " << this->m_tableau.name << " solver, " <<
1229 "at t = " << t_tmp << ", integration failed, adding substepping layer.");}
1230 h_tmp /= Real(2.0);
1231 continue;
1232 }
1233
1234 // Store independent variable (or time)
1235 t_tmp += h_tmp;
1236 }
1237
1238 // Store output states substepping solutions
1239 x_new = x_tmp;
1240 h_new = h_tmp;
1241 }
1242
1243 // Project intermediate solution on the invariants
1244 if (this->m_projection) {
1245 VectorN x_projected;
1246 if (this->project(x_new, t_old + h_new, x_projected)) {
1247 x_new = x_projected;
1248 return true;
1249 } else {
1250 return false;
1251 }
1252 } else {
1253 return true;
1254 }

1268 bool solve(VectorX const &t_mesh, VectorN const &ics, Solution<Real, N, M> &sol)
1269 {
1270 using Eigen::last;
1271
1272 // Instantiate output
1273 sol.resize(t_mesh.size());
1274
1275 // Store first step
1276 sol.t(0) = t_mesh(0);
1277 sol.x.col(0) = ics;
1278 sol.h.col(0) = this->m_system->h(ics, t_mesh(0));
1279
1280 // Update the current step
1281 Integer step{0};
1282 VectorN x_step{ics};
1283 Real t_step{t_mesh(0)}, h_step{t_mesh(1) - t_mesh(0)}, h_tmp_step{h_step}, h_new_step;
1284 bool mesh_point_bool, saturation_bool;
1285
1286 while (true) {
1287 // Integrate system
1288 if (!this->advance(sol.x.col(step), t_step, h_step, x_step, h_new_step)) {return false;}
1289
1290 // Update the current step
1291 t_step += h_step;
1292
1293 // Saturate the suggested timestep
1294 mesh_point_bool = std::abs(t_step - t_mesh(step+1)) < SQRT_EPSILON;
1295 saturation_bool = t_step + h_new_step > t_mesh(step+1) + SQRT_EPSILON;
1296 if (this->m_adaptive && this->m_tableau.is_embedded && !mesh_point_bool && saturation_bool) {
1297 h_tmp_step = h_new_step; // Used to store the previous step and keep the integration pace
1298 h_step = t_mesh(step+1) - t_step;
1299 } else {
1300 h_step = h_new_step;
1301 }
1302
1303 // Store solution if the step is a mesh point
1304 if (!this->m_adaptive || mesh_point_bool) {
1305 // Update temporaries
1306 step += 1;
1307 h_step = h_tmp_step;
1308
1309 // Update outputs
1310 sol.t(step) = t_step;
1311 sol.x.col(step) = x_step;
1312 sol.h.col(step) = this->m_system->h(x_step, t_step);
1313
1314 // Check if the current step is the last one
1315 if (std::abs(t_step - t_mesh(last)) < SQRT_EPSILON) {break;}
1316 }
1317 }
1318 return true;

1330 bool adaptive_solve(VectorX const &t_mesh, VectorN const &ics, Solution<Real, N, M> &sol)
1331 {
1332 using Eigen::all;
1333 using Eigen::last;
1334
1335 #define CMD "Sandals::RungeKutta::adaptive_solve(...): "
1336
1337 // Check if the adaptive method is enabled and the method is embedded
1338 if (!this->is_embedded()) {
1339 SANDALS_WARNING(CMD "the method is not embedded, using solve(...) method.");
1340 return this->solve(t_mesh, ics, sol);
1341 } else if (!this->m_adaptive) {
1342 SANDALS_WARNING(CMD "adaptive method is disabled, using solve(...) method.");
1343 return this->solve(t_mesh, ics, sol);
1344 }
1345
1346 // Instantiate output
1347 Real h_step{t_mesh(1) - t_mesh(0)}, h_new_step, scale{100.0};
1348 Real h_min{std::max(this->m_min_step, h_step/scale)}, h_max{scale*h_step};
1349 if (this->m_tableau.is_embedded) {
1350 Integer safety_length{static_cast<Integer>(std::ceil(std::abs(t_mesh(last) - t_mesh(0))/(2.0*h_min)))};
1351 sol.resize(safety_length);
1352 } else {
1353 sol.resize(t_mesh.size());
1354 }
1355
1356 // Store first step
1357 sol.t(0) = t_mesh(0);
1358 sol.x.col(0) = ics;

1360
1361 // Instantiate temporary variables
1362 Integer step{0};
1363 VectorN x_step{ics};
1364
1365 while (true) {
1366 // Integrate system
1367 this->advance(sol.x.col(step), sol.t(step), h_step, x_step, h_new_step);
1368
1369 // Saturate the suggested timestep
1370 if (this->m_adaptive && this->m_tableau.is_embedded) {
1371 h_step = std::max(std::min(h_new_step, h_max), h_min);
1372 }
1373
1374 SANDALS_ASSERT(step < sol.size(), CMD "safety length exceeded.");
1375
1376 // Store solution
1377 sol.t(step+1) = sol.t(step) + h_step;
1378 sol.x.col(step+1) = x_step;
1379 sol.h.col(step+1) = this->m_system->h(x_step, sol.t(step+1));
1380
1381 // Check if the current step is the last one
1382 if (sol.t(step+1) + h_step > t_mesh(last)) {break;}
1383
1384 // Update steps counter
1385 step += 1;
1386 }
1387
1388 // Resize the output
1389 sol.conservative_resize(step-1);
1390
1391 return true;

1405 bool project(VectorN const &x, Real t, VectorN &x_projected)
1406 {
1407 #define CMD "Sandals::RungeKutta::project(...): "
1408
1409 // Check if there are any constraints
1410 x_projected = x;
1411 if (M > Integer(0)) {
1412
1413 VectorM h;
1414 MatrixM Jh_x;
1415 VectorP b, x_step;
1416 MatrixP A;
1417 A.setZero();
1418 A.template block<N, N>(0, 0) = MatrixN::Identity();
1419 for (Integer k{0}; k < this->m_max_projection_iterations; ++k) {
1420
1421 /* Standard projection method
1422 [A] {x} = {b}
1423 / I Jh_x^T \ / dx \ / x_t - x_k \
1424 | | | | = | |
1425 \ Jh_x 0 / \ lambda / \ -h /
1426 */
1427
1428 // Evaluate the invariants vector and its Jacobian
1429 h = this->m_system->h(x_projected, t);
1430 Jh_x = this->m_system->Jh_x(x_projected, t);
1431
1432 // Check if the solution is found
1433 if (h.norm() < this->m_projection_tolerance) {return true;}
1434
1435 // Build the linear system
1436 A.template block<N, M>(0, N) = Jh_x.transpose();
1437 A.template block<M, N>(N, 0) = Jh_x;
1438 b.template head<N>() = x - x_projected;
1439 b.template tail<M>() = -h;
1440
1441 // Compute the solution of the linear system
1442 this->m_lu.compute(A);
1443 SANDALS_ASSERT(this->m_lu.rank() == N+M, CMD "singular Jacobian detected.");
1444 x_step = this->m_lu.solve(b);
1445
1446 // Check if the step is too small
1447 if (x_step.norm() < this->m_projection_tolerance * this->m_projection_tolerance) {return false;}
1448
1449 // Update the solution
1450 x_projected.noalias() += x_step(Eigen::seqN(0, N));
1451 }
1452 if (this->m_verbose) {SANDALS_WARNING(CMD "maximum number of iterations reached.");}
1453 return false;
1454 } else {
1455 return true;

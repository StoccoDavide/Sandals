565 bool explicit_step(VectorN const &x_k, Real t_k, Real t_step, VectorN &x_out, Real &t_adap) const
566 {
567 using Eigen::all;
568 using Eigen::seqN;
569
570 // Compute the K variables in the case of an explicit method and explicit system
571 Real t_i;
572 VectorN x_i;
573 MatrixK K;
574 for (Size i = 0; i < S; ++i)
575 {
576 t_i = t_k + this->m_tableau.c(i) * t_step;
577 x_i = x_k + K(all, seqN(0, i)) * this->m_tableau.A(i, seqN(0, i)).transpose();
578 K.col(i) = t_step * static_cast<Explicit<N> const *>(this->m_system.get())->f(x_i, t_i);
579 }
580 if (!K.allFinite()) {return false;}
581
582 // Perform the step and obtain the next state
583 x_out = x_k + K * this->m_tableau.b;
584
585 // Adapt next time step
586 if (this->m_adaptive_step && this->m_is_embedded) {
587 VectorN x_e = x_k + K * this->m_tableau.b_e;
588 t_adap = this->estimate_step(x_out, x_e, t_step);
589 }
590 return true;
591 }

606 void implicit_function(VectorN const &x_k, VectorK const &K_k, Real t_k, Real d_t, VectorK &fun) const
607 {
608 fun.setZero();
609
610 // Loop through each equation of the system
611 Real t_i;
612 VectorN x_i;
613 MatrixK K{K_k.reshaped(N, S)}; // FIXME: can i avoid reshaping?
614 MatrixK fun_tmp;
615 for (Size i = 0; i < S; ++i) {
616 t_i = t_k + this->m_tableau.c(i) * d_t;
617 x_i = x_k + K * this->m_tableau.A.row(i).transpose();
618 fun_tmp.col(i) = this->m_system->F(x_i, K.col(i) / d_t, t_i);
619 }
620 fun = fun_tmp.reshaped(N*S, 1);
621 }

645 void implicit_jacobian(VectorN const &x_k, VectorK const &K_k, Real t_k, Real d_t, MatrixJ &jac) const
646 {
647 using Eigen::all;
648 using Eigen::seqN;
649
650 jac.setZero();
651
652 // Loop through each equation of the system
653 MatrixK K{K_k.reshaped(N, S)};
654 Real t_i;
655 VectorN x_i, x_dot_i;
656 MatrixN JF_x, JF_x_dot;
657 auto idx = seqN(0, N), jdx = seqN(0, N);
658 for (Size i = 0; i < S; ++i) {
659 t_i = t_k + this->m_tableau.c(i) * d_t;
660 x_i = x_k + K * this->m_tableau.A.row(i).transpose();
661
662 // Compute the Jacobians with respect to x and x_dot
663 x_dot_i = K.col(i) / d_t;
664 JF_x = this->m_system->JF_x(x_i, x_dot_i, t_i);
665 JF_x_dot = this->m_system->JF_x_dot(x_i, x_dot_i, t_i);
666
667 // Derivative of F(x_i, K(:,i)/d_t, t_i)
668 idx = seqN(i*N, N);
669 for (Size j = 0; j < S; ++j) {
670 // Combine the Jacobians with respect to x and x_dot to obtain the
671 // Jacobian with respect to K
672 jdx = seqN(j*N, N);
673 if (i == j) {
674 jac(idx, jdx) = this->m_tableau.A(i,j) * JF_x + JF_x_dot / d_t;
675 } else {
676 jac(idx, jdx) = this->m_tableau.A(i,j) * JF_x;
677 }
678 }
679 }

752 bool implicit_step(VectorN const &x_k, Real t_k, Real d_t, VectorN &x_o, Real &d_t_star)
753 {
754 // Check if the solver converged
755 VectorK K;
756
757 if (!this->m_newton.solve(
758 [this, &x_k, t_k, d_t](VectorK const &K, VectorK &fun) {this->implicit_function(x_k, K, t_k, d_t, fun);},
759 [this, &x_k, t_k, d_t](VectorK const &K, MatrixJ &jac) {this->implicit_jacobian(x_k, K, t_k, d_t, jac);},
760 VectorK::Zero(), K)) {return false;}
761
762 // Perform the step and obtain x_k+1
763 x_o = x_k + K.reshaped(N, S) * this->m_tableau.b;
764
765 // Adapt next time step
766 if (this->m_adaptive_step && this->m_tableau.is_embedded) {
767 VectorN x_e{x_k + K.reshaped(N, S) * this->m_tableau.b_e};
768 d_t_star = this->estimate_step(x_o, x_e, d_t);
769 }
770 return true;
771 }

823 bool advance(VectorN const &x_k, Real t_k, Real d_t, VectorN &x_new, Real &d_t_star)
824 {
825 #define CMD "Indigo.RungeKutta.advance(...): "
826
827 // Check step size
828 SANDALS_ASSERT(d_t > Real(0.0), CMD "in " << this->m_tableau.name <<
829 "solver, d_t = "<< d_t << ", expected > 0.");
830
831 // If the integration step failed, try again with substepping
832 if (!this->step(x_k, t_k, d_t, x_new, d_t_star))
833 {
834 VectorN x_tmp = x_k;
835 Real t_tmp = t_k;
836 Real d_t_tmp = d_t / Real(2.0);
837
838 // Substepping logic
839 Size max_substeps = 10;
840 Size max_k = max_substeps * max_substeps;
841 Size k = 2;
842 Real d_t_star_tmp;
843 while (k > 0)
844 {
845 // Calculate the next time step with substepping logic
846 if (this->step(x_tmp, t_tmp, d_t_tmp, x_new, d_t_star_tmp)) {
847
848 // Accept the step
849 d_t_tmp = d_t_star_tmp;
850
851 // If substepping is enabled, double the step size
852 if (k > 0 && k < max_k) {
853 k = k - 1;
854 // If the substepping index is even, double the step size
855 if (k % 2 == 0) {
856 d_t_tmp = Real(2.0) * d_t_tmp;
857 if (this->m_verbose) {
858 SANDALS_WARNING(CMD "in " << this->m_tableau.name << " solver, at t = " <<
859 t_tmp << ", integration succedded disable one substepping layer.");
860 }
861 }
862 }
863
864 // Check the infinity norm of the solution
865 SANDALS_ASSERT(std::isfinite(x_tmp.maxCoeff()), CMD "in " <<
866 this->m_tableau.name << " solver, at t = " << t_tmp <<
867 ", ||x||_inf = inf, computation interrupted.");
868
869 } else {
870
871 // If the substepping index is too high, abort the integration
872 k = k + 2;
873 SANDALS_ASSERT(k < max_k, CMD "in " << this->m_tableau.name <<
874 " solver, at t = " << t_tmp << ", integration failed with d_t = " <<
875 d_t << ", aborting.");
876 return false;
877
878 // Otherwise, try again with a smaller step
879 if (this->m_verbose) {
880 SANDALS_WARNING(CMD "in " << this->m_tableau.name << " solver, at t = " <<
881 t_tmp << ", integration failed, adding substepping layer.");
882 }
883 d_t_tmp = d_t_tmp / Real(2.0);
884 continue;
885
886 }
887
888 // Store time solution
889 t_tmp = t_tmp + d_t_tmp;
890 }
891
892 // Store output states substepping solutions
893 x_new = x_tmp;
894 d_t_star = d_t_tmp;
895 return true;
896 } else {
897 return true;
898 }
899 #undef CMD
900 }

908 bool solve(VectorX const &t, VectorN const &ics, Solution &sol)
909 {
910 using Eigen::last;
911
912 // Instantiate output
913 sol.resize(t.size());
914
915 // Store first step
916 sol.time(0) = t(0);
917 sol.states.col(0) = ics;
918
919 // Update the current step
920 Size s{0};
921 VectorN x_s{ics};
922 Real t_s{t(0)};
923 Real d_t_s{t(1) - t(0)};
924 Real d_t_tmp{d_t_s}, d_t_star;
925 bool mesh_point_bool, saturation_bool;
926
927 while (true) {
928 // Integrate system
929 if (!this->advance(sol.states.col(s), t_s, d_t_s, x_s, d_t_star)) {return false;}
930
931 // Update the current step
932 t_s = t_s + d_t_s;
933
934 // Saturate the suggested timestep
935 mesh_point_bool = std::abs(t_s - t(s+1)) < SQRT_EPSILON;
936 saturation_bool = t_s + d_t_star > t(s+1) + SQRT_EPSILON;
937 if (this->m_adaptive_step && !mesh_point_bool && saturation_bool) {
938 d_t_tmp = d_t_star;
939 d_t_s = t(s+1) - t_s;
940 } else {
941 d_t_s = d_t_star;
942 }
943
944 // Store solution if the step is a mesh point
945 if (!this->m_adaptive_step || mesh_point_bool) {
946 // Update temporaries
947 s += 1;
948 d_t_s = d_t_tmp;
949
950 // Update outputs
951 sol.time(s) = t_s;
952 sol.states.col(s) = x_s;
953
954 // Check if the current step is the last one
955 if (std::abs(t_s - t(last)) < SQRT_EPSILON) {break;}
956 }
957 }
958 return true;
959 }

560 bool explicit_step(VectorN const &x_k, Real t_k, Real t_step, VectorN &x_out, Real &t_adap) const
561 {
562 using Eigen::all;
563 using Eigen::seqN;
564
565 // Compute the K variables in the case of an explicit method and explicit system
566 Real t_i;
567 VectorN x_i;
568 MatrixK K;
569 for (Size i = 0; i < S; ++i)
570 {
571 t_i = t_k + this->m_tableau.c(i) * t_step;
572 x_i = x_k + K(all, seqN(0, i)) * this->m_tableau.A(i, seqN(0, i)).transpose();
573 K.col(i) = t_step * static_cast<Explicit<N> const *>(this->m_system.get())->f(x_i, t_i);
574 }
575 if (!K.allFinite()) {return false;}
576
577 // Perform the step and obtain the next state
578 x_out = x_k + K * this->m_tableau.b;
579
580 // Adapt next time step
581 if (this->m_adaptive_step && this->m_is_embedded) {
582 VectorN x_e = x_k + K * this->m_tableau.b_e;
583 t_adap = this->estimate_step(x_out, x_e, t_step);
584 }
585 return true;
586 }

601 void implicit_function(VectorN const &x_k, VectorK const &K_k, Real t_k, Real d_t, VectorK &fun) const
602 {
603 fun.setZero();
604
605 // Loop through each equation of the system
606 Real t_i;
607 VectorN x_i;
608 MatrixK K{K_k.reshaped(N, S)}; // FIXME: can i avoid reshaping?
609 MatrixK fun_tmp;
610 for (Size i = 0; i < S; ++i) {
611 t_i = t_k + this->m_tableau.c(i) * d_t;
612 x_i = x_k + K * this->m_tableau.A.row(i).transpose();
613 fun_tmp.col(i) = this->m_system->F(x_i, K.col(i) / d_t, t_i);
614 }
615 fun = fun_tmp.reshaped(N*S, 1);
616 }

640 void implicit_jacobian(VectorN const &x_k, VectorK const &K_k, Real t_k, Real d_t, MatrixJ &jac) const
641 {
642 using Eigen::all;
643 using Eigen::seqN;
644
645 jac.setZero();
646
647 // Loop through each equation of the system
648 MatrixK K{K_k.reshaped(N, S)};
649 Real t_i;
650 VectorN x_i, x_dot_i;
651 MatrixN JF_x, JF_x_dot;
652 auto idx = seqN(0, N), jdx = seqN(0, N);
653 for (Size i = 0; i < S; ++i) {
654 t_i = t_k + this->m_tableau.c(i) * d_t;
655 x_i = x_k + K * this->m_tableau.A.row(i).transpose();
656
657 // Compute the Jacobians with respect to x and x_dot
658 x_dot_i = K.col(i) / d_t;
659 JF_x = this->m_system->JF_x(x_i, x_dot_i, t_i);
660 JF_x_dot = this->m_system->JF_x_dot(x_i, x_dot_i, t_i);
661
662 // Derivative of F(x_i, K(:,i)/d_t, t_i)
663 idx = seqN(i*N, N);
664 for (Size j = 0; j < S; ++j) {
665 // Combine the Jacobians with respect to x and x_dot to obtain the
666 // Jacobian with respect to K
667 jdx = seqN(j*N, N);
668 if (i == j) {
669 jac(idx, jdx) = this->m_tableau.A(i,j) * JF_x + JF_x_dot / d_t;
670 } else {
671 jac(idx, jdx) = this->m_tableau.A(i,j) * JF_x;
672 }
673 }
674 }

747 bool implicit_step(VectorN const &x_k, Real t_k, Real d_t, VectorN &x_o, Real &d_t_star)
748 {
749 // Check if the solver converged
750 VectorK K;
751
752 if (!this->m_newton.solve(
753 [this, &x_k, t_k, d_t](VectorK const &K, VectorK &fun) {this->implicit_function(x_k, K, t_k, d_t, fun);},
754 [this, &x_k, t_k, d_t](VectorK const &K, MatrixJ &jac) {this->implicit_jacobian(x_k, K, t_k, d_t, jac);},
755 VectorK::Zero(), K)) {return false;}
756
757 // Perform the step and obtain x_k+1
758 x_o = x_k + K.reshaped(N, S) * this->m_tableau.b;
759
760 // Adapt next time step
761 if (this->m_adaptive_step && this->m_tableau.is_embedded) {
762 VectorN x_e{x_k + K.reshaped(N, S) * this->m_tableau.b_e};
763 d_t_star = this->estimate_step(x_o, x_e, d_t);
764 }
765 return true;
766 }

818 bool advance(VectorN const &x_k, Real t_k, Real d_t, VectorN &x_new, Real &d_t_star)
819 {
820 #define CMD "Indigo.RungeKutta.advance(...): "
821
822 // Check step size
823 SANDALS_ASSERT(d_t > Real(0.0), CMD "in " << this->m_tableau.name <<
824 "solver, d_t = "<< d_t << ", expected > 0.");
825
826 // If the integration step failed, try again with substepping
827 if (!this->step(x_k, t_k, d_t, x_new, d_t_star))
828 {
829 VectorN x_tmp = x_k;
830 Real t_tmp = t_k;
831 Real d_t_tmp = d_t / Real(2.0);
832
833 // Substepping logic
834 Size max_substeps = 10;
835 Size max_k = max_substeps * max_substeps;
836 Size k = 2;
837 Real d_t_star_tmp;
838 while (k > 0)
839 {
840 // Calculate the next time step with substepping logic
841 if (this->step(x_tmp, t_tmp, d_t_tmp, x_new, d_t_star_tmp)) {
842
843 // Accept the step
844 d_t_tmp = d_t_star_tmp;
845
846 // If substepping is enabled, double the step size
847 if (k > 0 && k < max_k) {
848 k = k - 1;
849 // If the substepping index is even, double the step size
850 if (k % 2 == 0) {
851 d_t_tmp = Real(2.0) * d_t_tmp;
852 if (this->m_verbose) {
853 SANDALS_WARNING(CMD "in " << this->m_tableau.name << " solver, at t = " <<
854 t_tmp << ", integration succedded disable one substepping layer.");
855 }
856 }
857 }
858
859 // Check the infinity norm of the solution
860 SANDALS_ASSERT(std::isfinite(x_tmp.maxCoeff()), CMD "in " <<
861 this->m_tableau.name << " solver, at t = " << t_tmp <<
862 ", ||x||_inf = inf, computation interrupted.");
863
864 } else {
865
866 // If the substepping index is too high, abort the integration
867 k = k + 2;
868 SANDALS_ASSERT(k < max_k, CMD "in " << this->m_tableau.name <<
869 " solver, at t = " << t_tmp << ", integration failed with d_t = " <<
870 d_t << ", aborting.");
871 return false;
872
873 // Otherwise, try again with a smaller step
874 if (this->m_verbose) {
875 SANDALS_WARNING(CMD "in " << this->m_tableau.name << " solver, at t = " <<
876 t_tmp << ", integration failed, adding substepping layer.");
877 }
878 d_t_tmp = d_t_tmp / Real(2.0);
879 continue;
880
881 }
882
883 // Store time solution
884 t_tmp = t_tmp + d_t_tmp;
885 }
886
887 // Store output states substepping solutions
888 x_new = x_tmp;
889 d_t_star = d_t_tmp;
890 return true;
891 } else {
892 return true;
893 }
894 #undef CMD
895 }

907 bool solve(VectorD const &t, VectorN const &ics, Solution &sol)
908 {
909 using Eigen::last;
910
911 // Instantiate output
912 sol.resize(t.size());
913
914 // Store first step
915 sol.time(0) = t(0);
916 sol.states.col(0) = ics;
917
918 // Update the current step
919 Size s{0};
920 VectorN x_s{ics};
921 Real t_s{t(0)};
922 Real d_t_s{t(1) - t(0)};
923 Real d_t_tmp{d_t_s}, d_t_star;
924 bool mesh_point_bool, saturation_bool;
925
926 while (true) {
927 // Integrate system
928 if (!this->advance(sol.states.col(s), t_s, d_t_s, x_s, d_t_star)) {return false;}
929
930 // Update the current step
931 t_s = t_s + d_t_s;
932
933 // Saturate the suggested timestep
934 mesh_point_bool = std::abs(t_s - t(s+1)) < SQRT_EPSILON;
935 saturation_bool = t_s + d_t_star > t(s+1) + SQRT_EPSILON;
936 if (this->m_adaptive_step && !mesh_point_bool && saturation_bool) {
937 d_t_tmp = d_t_star;
938 d_t_s = t(s+1) - t_s;
939 } else {
940 d_t_s = d_t_star;
941 }
942
943 // Store solution if the step is a mesh point
944 if (!this->m_adaptive_step || mesh_point_bool) {
945 // Update temporaries
946 s += 1;
947 d_t_s = d_t_tmp;
948
949 // Update outputs
950 sol.time(s) = t_s;
951 sol.states.col(s) = x_s;
952
953 // Check if the current step is the last one
954 if (std::abs(t_s - t(last)) < SQRT_EPSILON) {break;}
955 }
956 }
957 return true;
958 }

971 using Eigen::last;
972
973 #define CMD "Sandals::RungeKutta::adaptive_solve(...): "
974

977 Real d_t{t(1) - t(0)}, d_t_star;

979 std::cout << "d_t = " << d_t << std::endl;
980

982 Real t_max{scale*d_t};

985 safety_length = std::ceil(Real(1.5)/this->m_min_sft_fac) * t.size();
986 sol.resize(safety_length);

988 sol.resize(t.size());
989 }
990
991 // Store first step
992 sol.time(0) = t(0);
993 sol.states.col(0) = ics;
994
995 // Instantiate temporary variables
996 Size s{0};
997 VectorN x_s{ics};
998
999 while (true) {
1000 // Integrate system
1001 this->advance(sol.states.col(s), sol.time(s), d_t, x_s, d_t_star);
1002
1003 // Saturate the suggested timestep
1004 if (this->m_adaptive_step && this->m_tableau.is_embedded) {
1005 d_t = std::max(std::min(d_t_star, t_max), t_min);
1006 }
1007 std::cout << "d_t = " << d_t_star << std::endl;
1008
1009 SANDALS_ASSERT(s < t.size(), CMD "safety length exceeded.");
1010
1011 // Store solution
1012 sol.time(s+1) = sol.time(s) + d_t;
1013 sol.states.col(s+1) = x_s;
1014
1015 // Check if the current step is the last one
1016 if (sol.time(s+1) + d_t > t(last)) {break;}
1017
1018 // Update steps counter
1019 s += 1;
1020 }
1021
1022 // Resize the output
1023 sol.conservative_resize(s-1);
1024

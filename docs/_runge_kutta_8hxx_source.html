514 bool erk_step(VectorN const &x_k, Real t_k, Real t_step, VectorN &x_out, Real &t_adap) const
515 {
516 using Eigen::all;
517 using Eigen::seqN;
518
519 // Compute the K variables in the case of an explicit method and explicit system
520 Real t_i;
521 VectorN x_i;
522 MatrixK K;
523 for (Size i = 0; i < S; ++i)
524 {
525 t_i = t_k + this->m_tableau.c(i) * t_step;
526 x_i = x_k + K(all, seqN(0, i)) * this->m_tableau.A(i, seqN(0, i)).transpose();
527 K.col(i) = t_step * static_cast<Explicit<N> const *>(this->m_system.get())->f(x_i, t_i);
528 }
529 if (!K.allFinite()) {return false;}
530
531 // Perform the step and obtain the next state
532 x_out = x_k + K * this->m_tableau.b;
533
534 // Adapt next time step
535 if (this->m_adaptive_step && this->m_is_embedded) {
536 VectorN x_e = x_k + K * this->m_tableau.b_e;
537 t_adap = this->estimate_step(x_out, x_e, t_step);
538 }
539 return true;
540 }

584 void irk_function(VectorN const &x_k, VectorK const &K_k, Real t_k, Real d_t, VectorK &fun) const
585 {
586 fun.setZero();
587
588 // Loop through each equation of the system
589 Real t_i;
590 VectorN x_i;
591 MatrixK K{K_k.reshaped(N, S)}; // FIXME: can i avoid reshaping?
592 MatrixK fun_tmp;
593 for (Size i = 0; i < S; ++i) {
594 t_i = t_k + this->m_tableau.c(i) * d_t;
595 x_i = x_k + K * this->m_tableau.A.row(i).transpose();
596 fun_tmp.col(i) = this->m_system->F(x_i, K.col(i) / d_t, t_i);
597 }
598 fun = fun_tmp.reshaped(N*S, 1);
599 }

623 void irk_jacobian(VectorN const &x_k, VectorK const &K_k, Real t_k, Real d_t, MatrixJ &jac) const
624 {
625 using Eigen::all;
626 using Eigen::seqN;
627
628 jac.setZero();
629
630 // Loop through each equation of the system
631 MatrixK K{K_k.reshaped(N, S)};
632 Real t_i;
633 VectorN x_i, x_dot_i;
634 MatrixN JF_x, JF_x_dot;
635 auto idx = seqN(0, N), jdx = seqN(0, N);
636 for (Size i = 0; i < S; ++i) {
637 t_i = t_k + this->m_tableau.c(i) * d_t;
638 x_i = x_k + K * this->m_tableau.A.row(i).transpose();
639
640 // Compute the Jacobians with respect to x and x_dot
641 x_dot_i = K.col(i) / d_t;
642 JF_x = this->m_system->JF_x(x_i, x_dot_i, t_i);
643 JF_x_dot = this->m_system->JF_x_dot(x_i, x_dot_i, t_i);
644
645 // Derivative of F(x_i, K(:,i)/d_t, t_i)
646 idx = seqN(i*N, N);
647 for (Size j = 0; j < S; ++j) {
648 // Combine the Jacobians with respect to x and x_dot to obtain the
649 // Jacobian with respect to K
650 jdx = seqN(j*N, N);
651 if (i == j) {
652 jac(idx, jdx) = this->m_tableau.A(i,j) * JF_x + JF_x_dot / d_t;
653 } else {
654 jac(idx, jdx) = this->m_tableau.A(i,j) * JF_x;
655 }
656 }
657 }

728 bool irk_step(VectorN const &x_k, Real t_k, Real d_t, VectorN &x_o, Real &d_t_star)
729 {
730 // Check if the solver converged
731 VectorK K;
732
733 if (!this->m_newton.solve(
734 [this, &x_k, t_k, d_t](VectorK const &K, VectorK &fun) {this->irk_function(x_k, K, t_k, d_t, fun);},
735 [this, &x_k, t_k, d_t](VectorK const &K, MatrixJ &jac) {this->irk_jacobian(x_k, K, t_k, d_t, jac);},
736 VectorK::Zero(), K)) {return false;}
737
738 // Perform the step and obtain x_k+1
739 x_o = x_k + K.reshaped(N, S) * this->m_tableau.b;
740
741 // Adapt next time step
742 if (this->m_adaptive_step && this->m_tableau.is_embedded) {
743 VectorN x_e{x_k + K.reshaped(N, S) * this->m_tableau.b_e};
744 d_t_star = this->estimate_step(x_o, x_e, d_t);
745 }
746 return true;
747 }

794 bool advance(VectorN const &x_k, Real t_k, Real d_t, VectorN &x_new, Real &d_t_star)
795 {
796 #define CMD "Indigo.RungeKutta.advance(...): "
797
798 // Check step size
799 SANDALS_ASSERT(d_t > Real(0.0), CMD "in " << this->m_tableau.name <<
800 "solver, d_t = "<< d_t << ", expected > 0.");
801
802 // If the integration step failed, try again with substepping
803 if (!this->step(x_k, t_k, d_t, x_new, d_t_star))
804 {
805 VectorN x_tmp = x_k;
806 Real t_tmp = t_k;
807 Real d_t_tmp = d_t / Real(2.0);
808
809 // Substepping logic
810 Size max_substeps = 10;
811 Size max_k = max_substeps * max_substeps;
812 Size k = 2;
813 Real d_t_star_tmp;
814 while (k > 0)
815 {
816 // Calculate the next time step with substepping logic
817 if (this->step(x_tmp, t_tmp, d_t_tmp, x_new, d_t_star_tmp)) {
818
819 // Accept the step
820 d_t_tmp = d_t_star_tmp;
821
822 // If substepping is enabled, double the step size
823 if (k > 0 && k < max_k) {
824 k = k - 1;
825 // If the substepping index is even, double the step size
826 if (k % 2 == 0) {
827 d_t_tmp = Real(2.0) * d_t_tmp;
828 if (this->m_verbose) {
829 SANDALS_WARNING(CMD "in " << this->m_tableau.name << " solver, at t = " <<
830 t_tmp << ", integration succedded disable one substepping layer.");
831 }
832 }
833 }
834
835 // Check the infinity norm of the solution
836 SANDALS_ASSERT(std::isfinite(x_tmp.maxCoeff()), CMD "in " <<
837 this->m_tableau.name << " solver, at t = " << t_tmp <<
838 ", ||x||_inf = inf, computation interrupted.");
839
840 } else {
841
842 // If the substepping index is too high, abort the integration
843 k = k + 2;
844 SANDALS_ASSERT(k < max_k, CMD "in " << this->m_tableau.name <<
845 " solver, at t = " << t_tmp << ", integration failed with d_t = " <<
846 d_t << ", aborting.");
847 return false;
848
849 // Otherwise, try again with a smaller step
850 if (this->m_verbose) {
851 SANDALS_WARNING(CMD "in " << this->m_tableau.name << " solver, at t = " <<
852 t_tmp << ", integration failed, adding substepping layer.");
853 }
854 d_t_tmp = d_t_tmp / Real(2.0);
855 continue;
856
857 }
858
859 // Store time solution
860 t_tmp = t_tmp + d_t_tmp;
861 }
862
863 // Store output states substepping solutions
864 x_new = x_tmp;
865 d_t_star = d_t_tmp;
866 return true;
867 } else {
868 return true;
869 }
870 #undef CMD
871 }

879 bool solve(VectorX const &t, VectorN const &ics, Solution &sol)
880 {
881 using Eigen::last;
882
883 // Instantiate output
884 sol.resize(t.size());
885
886 // Store first step
887 sol.time(0) = t(0);
888 sol.states.col(0) = ics;
889
890 // Update the current step
891 Size s{0};
892 VectorN x_s{ics};
893 Real t_s{t(0)};
894 Real d_t_s{t(1) - t(0)};
895 Real d_t_tmp{d_t_s}, d_t_star;
896 bool mesh_point_bool, saturation_bool;
897
898 while (true) {
899 // Integrate system
900 if (!this->advance(sol.states.col(s), t_s, d_t_s, x_s, d_t_star)) {return false;}
901
902 // Update the current step
903 t_s = t_s + d_t_s;
904
905 // Saturate the suggested timestep
906 mesh_point_bool = std::abs(t_s - t(s+1)) < SQRT_EPSILON;
907 saturation_bool = t_s + d_t_star > t(s+1) + SQRT_EPSILON;
908 if (this->m_adaptive_step && !mesh_point_bool && saturation_bool) {
909 d_t_tmp = d_t_star;
910 d_t_s = t(s+1) - t_s;
911 } else {
912 d_t_s = d_t_star;
913 }
914
915 // Store solution if the step is a mesh point
916 if (!this->m_adaptive_step || mesh_point_bool) {
917 // Update temporaries
918 s += 1;
919 d_t_s = d_t_tmp;
920
921 // Update outputs
922 sol.time(s) = t_s;
923 sol.states.col(s) = x_s;
924
925 // Check if the current step is the last one
926 if (std::abs(t_s - t(last)) < SQRT_EPSILON) {break;}
927 }
928 }
929 return true;
930 }

939 bool adaptive_solve(VectorX const &t, VectorN const &ics, Solution &sol)
940 {
941 using Eigen::last;
942
943 #define CMD "Sandals::RungeKutta::adaptive_solve(...): "
944
945 // Instantiate output
946 this->enable_adaptive_step();
947 Real d_t{t(1) - t(0)}, d_t_star;
948 Real scale{100.0};
949 std::cout << "d_t = " << d_t << std::endl;
950
951 Real t_min{std::max(this->m_min_step, d_t/scale)};
952 Real t_max{scale*d_t};
953 Size safety_length;
954 if (this->m_tableau.is_embedded) {
955 safety_length = std::ceil(Real(1.5)/this->m_min_safety_factor) * t.size();
956 sol.resize(safety_length);
957 } else {
958 sol.resize(t.size());
959 }
960
961 // Store first step
962 sol.time(0) = t(0);
963 sol.states.col(0) = ics;
964
965 // Instantiate temporary variables
966 Size s{0};
967 VectorN x_s{ics};
968
969 while (true) {
970 // Integrate system
971 this->advance(sol.states.col(s), sol.time(s), d_t, x_s, d_t_star);
972
973 // Saturate the suggested timestep
974 if (this->m_adaptive_step && this->m_tableau.is_embedded) {
975 d_t = std::max(std::min(d_t_star, t_max), t_min);
976 }
977 std::cout << "d_t = " << d_t_star << std::endl;
978
979 SANDALS_ASSERT(s < t.size(), CMD "safety length exceeded.");
980
981 // Store solution
982 sol.time(s+1) = sol.time(s) + d_t;
983 sol.states.col(s+1) = x_s;
984
985 // Check if the current step is the last one
986 if (sol.time(s+1) + d_t > t(last)) {break;}
987
988 // Update steps counter
989 s += 1;
990 }
991
992 // Resize the output
993 sol.conservative_resize(s-1);
994

456 bool erk_explicit_step(VectorN const &x_old, Real t_old, Real h_old, VectorN &x_new, Real &h_new) const
457 {
458 using Eigen::all;
459 using Eigen::seqN;
460
461 // Compute the K variables in the case of an explicit method and explicit system
462 Real t_node;
463 VectorN x_node;
464 MatrixK K;
465 for (Size i = 0; i < S; ++i)
466 {
467 t_node = t_old + this->m_tableau.c(i) * h_old;
468 x_node = x_old + K(all, seqN(0, i)) * this->m_tableau.A(i, seqN(0, i)).transpose();
469 K.col(i) = h_old * static_cast<Explicit<N> const *>(this->m_system.get())->f(x_node, t_node);
470 }
471 if (!K.allFinite()) {return false;}
472
473 // Perform the step and obtain the next state
474 x_new = x_old + K * this->m_tableau.b;
475
476 // Adapt next time step
477 if (this->m_adaptive_step && this->m_tableau.is_embedded) {
478 VectorN x_emb = x_old + K * this->m_tableau.b_e;
479 h_new = this->estimate_step(x_new, x_emb, h_old);
480 }
481 return true;
482 }

535 void erk_implicit_jacobian(Size n, VectorN const &x, Real t, Real h, VectorN const &K, MatrixS &jac) const
536 {
537 using Eigen::all;
538 using Eigen::seqN;
539
540 jac.setZero();
541
542 // Loop through each equation of the system
543 MatrixK K_mat{K.reshaped(N, S)};
544 Real t_node;
545 VectorN x_node, x_dot_node;
546 MatrixN JF_x, JF_x_dot;
547 auto idx = seqN(0, N), jdx = seqN(0, N);
548 for (Size i = 0; i < n; ++i) {
549 t_node = t + h * this->m_tableau.c(i);
550 x_node = x + K_mat(all, seqN(0, i)) * this->m_tableau.A.row(i).transpose();
551
552 // Compute the Jacobians with respect to x and x_dot
553 x_dot_node = K_mat.col(i)/h;
554 JF_x = this->m_system->JF_x(x_node, x_dot_node, t_node);
555 JF_x_dot = this->m_system->JF_x_dot(x_node, x_dot_node, t_node);
556
557 // Derivative of F(x_node, K(:,i)/h, t_node)
558 idx = seqN(i*N, N);
559 for (Size j = 0; j < S; ++j) {
560 // Combine the Jacobians with respect to x and x_dot to obtain the
561 // Jacobian with respect to K
562 jdx = seqN(j*N, N);
563 if (i == n) {
564 jac(idx, jdx) = this->m_tableau.A(i,j) * JF_x + JF_x_dot / h;
565 } else {
566 jac(idx, jdx) = this->m_tableau.A(i,j) * JF_x;
567 }
568 }
569 }
570 }

847 bool advance(VectorN const &x_k, Real t_k, Real d_t, VectorN &x_new, Real &d_t_star)
848 {
849 #define CMD "Indigo.RungeKutta.advance(...): "
850
851 // Check step size
852 SANDALS_ASSERT(d_t > Real(0.0), CMD "in " << this->m_tableau.name <<
853 "solver, d_t = "<< d_t << ", expected > 0.");
854
855 // If the integration step failed, try again with substepping
856 if (!this->step(x_k, t_k, d_t, x_new, d_t_star))
857 {
858 VectorN x_tmp = x_k;
859 Real t_tmp = t_k;
860 Real d_t_tmp = d_t / Real(2.0);
861
862 // Substepping logic
863 Size max_substeps = 10;
864 Size max_k = max_substeps * max_substeps;
865 Size k = 2;
866 Real d_t_star_tmp;
867 while (k > 0)
868 {
869 // Calculate the next time step with substepping logic
870 if (this->step(x_tmp, t_tmp, d_t_tmp, x_new, d_t_star_tmp)) {
871
872 // Accept the step
873 d_t_tmp = d_t_star_tmp;
874
875 // If substepping is enabled, double the step size
876 if (k > 0 && k < max_k) {
877 k = k - 1;
878 // If the substepping index is even, double the step size
879 if (k % 2 == 0) {
880 d_t_tmp = Real(2.0) * d_t_tmp;
881 if (this->m_verbose) {
882 SANDALS_WARNING(CMD "in " << this->m_tableau.name << " solver, at t = " <<
883 t_tmp << ", integration succedded disable one substepping layer.");
884 }
885 }
886 }
887
888 // Check the infinity norm of the solution
889 SANDALS_ASSERT(std::isfinite(x_tmp.maxCoeff()), CMD "in " <<
890 this->m_tableau.name << " solver, at t = " << t_tmp <<
891 ", ||x||_inf = inf, computation interrupted.");
892
893 } else {
894
895 // If the substepping index is too high, abort the integration
896 k = k + 2;
897 SANDALS_ASSERT(k < max_k, CMD "in " << this->m_tableau.name <<
898 " solver, at t = " << t_tmp << ", integration failed with d_t = " <<
899 d_t << ", aborting.");
900 return false;
901
902 // Otherwise, try again with a smaller step
903 if (this->m_verbose) {
904 SANDALS_WARNING(CMD "in " << this->m_tableau.name << " solver, at t = " <<
905 t_tmp << ", integration failed, adding substepping layer.");
906 }
907 d_t_tmp = d_t_tmp / Real(2.0);
908 continue;
909
910 }
911
912 // Store time solution
913 t_tmp = t_tmp + d_t_tmp;
914 }
915
916 // Store output states substepping solutions
917 x_new = x_tmp;
918 d_t_star = d_t_tmp;
919 return true;
920 } else {

932 bool solve(VectorX const &t, VectorN const &ics, Solution &sol)
933 {
934 using Eigen::last;
935
936 // Instantiate output
937 sol.resize(t.size());
938
939 // Store first step
940 sol.time(0) = t(0);
941 sol.states.col(0) = ics;
942
943 // Update the current step
944 Size s{0};
945 VectorN x_s{ics};
946 Real t_s{t(0)};
947 Real d_t_s{t(1) - t(0)};
948 Real d_t_tmp{d_t_s}, d_t_star;
949 bool mesh_point_bool, saturation_bool;
950
951 while (true) {
952 // Integrate system
953 if (!this->advance(sol.states.col(s), t_s, d_t_s, x_s, d_t_star)) {return false;}
954
955 // Update the current step
956 t_s = t_s + d_t_s;
957
958 // Saturate the suggested timestep
959 mesh_point_bool = std::abs(t_s - t(s+1)) < SQRT_EPSILON;
960 saturation_bool = t_s + d_t_star > t(s+1) + SQRT_EPSILON;
961 if (this->m_adaptive_step && !mesh_point_bool && saturation_bool) {
962 d_t_tmp = d_t_star;
963 d_t_s = t(s+1) - t_s;
964 } else {
965 d_t_s = d_t_star;
966 }
967
968 // Store solution if the step is a mesh point
969 if (!this->m_adaptive_step || mesh_point_bool) {
970 // Update temporaries
971 s += 1;
972 d_t_s = d_t_tmp;
973
974 // Update outputs
975 sol.time(s) = t_s;
976 sol.states.col(s) = x_s;
977
978 // Check if the current step is the last one
979 if (std::abs(t_s - t(last)) < SQRT_EPSILON) {break;}
980 }
981 }
982 return true;
983 }

992 bool adaptive_solve(VectorX const &t, VectorN const &ics, Solution &sol)
993 {
994 using Eigen::last;
995
996 #define CMD "Sandals::RungeKutta::adaptive_solve(...): "
997
998 // Instantiate output
999 this->enable_adaptive_step();
1000 Real d_t{t(1) - t(0)}, d_t_star;
1001 Real scale{100.0};
1002
1003 Real t_min{std::max(this->m_min_step, d_t/scale)};
1004 Real t_max{scale*d_t};
1005 Size safety_length;
1006 if (this->m_tableau.is_embedded) {
1007 safety_length = std::ceil(Real(1.5)/this->m_min_safety_factor) * t.size();
1008 sol.resize(safety_length);
1009 } else {
1010 sol.resize(t.size());
1011 }
1012
1013 // Store first step
1014 sol.time(0) = t(0);
1015 sol.states.col(0) = ics;
1016
1017 // Instantiate temporary variables
1018 Size s{0};
1019 VectorN x_s{ics};
1020
1021 while (true) {
1022 // Integrate system
1023 this->advance(sol.states.col(s), sol.time(s), d_t, x_s, d_t_star);
1024
1025 // Saturate the suggested timestep
1026 if (this->m_adaptive_step && this->m_tableau.is_embedded) {
1027 d_t = std::max(std::min(d_t_star, t_max), t_min);
1028 }
1029
1030 SANDALS_ASSERT(s < t.size(), CMD "safety length exceeded.");
1031
1032 // Store solution
1033 sol.time(s+1) = sol.time(s) + d_t;
1034 sol.states.col(s+1) = x_s;
1035
1036 // Check if the current step is the last one
1037 if (sol.time(s+1) + d_t > t(last)) {break;}
1038
1039 // Update steps counter
1040 s += 1;
1041 }
1042
1043 // Resize the output
1044 sol.conservative_resize(s-1);
1045
1046 return true;

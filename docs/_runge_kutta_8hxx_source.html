482 {
483 return h_k * std::min(this->m_max_safety_factor, std::max(this->m_min_safety_factor,

486 )).array().abs().maxCoeff()));

531 bool erk_explicit_step(VectorN const &x_old, Real t_old, Real h_old, VectorN &x_new, Real &h_new) const
532 {
533 using Eigen::all;
534 using Eigen::seqN;
535
536 // Compute the K variables in the case of an explicit method and explicit system
537 VectorN x_node;
538 MatrixK K;
539 for (Size i{0}; i < S; ++i) {
540 x_node = x_old + K(all, seqN(0, i)) * this->m_tableau.A(i, seqN(0, i)).transpose();
541 if (!this->m_reverse) {
542 K.col(i) = h_old * static_cast<Explicit<N, M> const *>(this->m_system.get())->f(x_node, t_old + h_old*this->m_tableau.c(i));
543 } else {
544 K.col(i) = h_old * static_cast<Explicit<N, M> const *>(this->m_system.get())->f_reverse(x_node, t_old + h_old*this->m_tableau.c(i));
545 }
546 }
547 if (!K.allFinite()) {return false;}
548
549 // Perform the step and obtain the next state
550 x_new = x_old + K * this->m_tableau.b;
551
552 // Adapt next step
553 if (this->m_adaptive && this->m_tableau.is_embedded) {
554 VectorN x_emb = x_old + K * this->m_tableau.b_e;
555 h_new = this->estimate_step(x_new, x_emb, h_old);
556 }
557 return true;
558 }

714 void irk_function(VectorN const &x, Real t, Real h, VectorK const &K, VectorK &fun) const
715 {
716 VectorN x_node;
717 MatrixK K_mat{K.reshaped(N, S)};
718 MatrixK fun_mat;
719 for (Size i{0}; i < S; ++i) {
720 x_node = x + K_mat * this->m_tableau.A.row(i).transpose();
721 if (!this->m_reverse) {
722 fun_mat.col(i) = this->m_system->F(x_node, K_mat.col(i)/h, t + h * this->m_tableau.c(i));
723 } else {
724 fun_mat.col(i) = this->m_system->F_reverse(x_node, K_mat.col(i)/h, t + h * this->m_tableau.c(i));
725 }
726 }
727 fun = fun_mat.reshaped(N*S, 1);
728 }

766 void irk_jacobian(VectorN const &x, Real t, Real h, VectorK const &K, MatrixJ &jac) const
767 {
768 using Eigen::seqN;
769
770 // Reset the Jacobian matrix
771 jac.setZero();
772
773 // Loop through each equation of the system
774 MatrixK K_mat{K.reshaped(N, S)};
775 Real t_node;
776 VectorN x_node, x_dot_node;
777 MatrixN JF_x, JF_x_dot;
778 auto idx = seqN(0, N), jdx = seqN(0, N);
779 for (Size i{0}; i < S; ++i) {
780 t_node = t + h * this->m_tableau.c(i);
781 x_node = x + K_mat * this->m_tableau.A.row(i).transpose();
782
783 // Compute the Jacobians with respect to x and x_dot
784 x_dot_node = K_mat.col(i) / h;
785 if (!this->m_reverse){
786 JF_x = this->m_system->JF_x(x_node, x_dot_node, t_node);
787 JF_x_dot = this->m_system->JF_x_dot(x_node, x_dot_node, t_node);
788 } else {
789 JF_x = this->m_system->JF_x_reverse(x_node, x_dot_node, t_node);
790 JF_x_dot = this->m_system->JF_x_dot_reverse(x_node, x_dot_node, t_node);
791 }
792
793 // Combine the Jacobians with respect to x and x_dot to obtain the Jacobian with respect to K
794 idx = seqN(i*N, N);
795 for (Size j{0}; j < S; ++j) {
796 jdx = seqN(j*N, N);
797 if (i == j) {
798 jac(idx, jdx) = this->m_tableau.A(i,j) * JF_x + JF_x_dot / h;
799 } else {
800 jac(idx, jdx) = this->m_tableau.A(i,j) * JF_x;
801 }
802 }
803 }
804 }

989 bool step(VectorN const &x_old, Real t_old, Real h_old, VectorN &x_new, Real &h_new)
990 {
991 #define CMD "Sandals::RungeKutta::step(...): "
992
993 SANDALS_ASSERT(this->m_system->in_domain(x_old, t_old), CMD "in " << this->m_tableau.name <<
994 " solver, at t = " << t_old << ", x = " << x_old.transpose() << ", system out of domain.");
995
996 if (this->is_erk() && this->m_system->is_explicit()) {
997 return this->erk_explicit_step(x_old, t_old, h_old, x_new, h_new);
998 } else if (this->is_erk() && this->m_system->is_implicit()) {
999 return this->erk_implicit_step(x_old, t_old, h_old, x_new, h_new);
1000 } else if (this->is_dirk()) {
1001 return this->dirk_step(x_old, t_old, h_old, x_new, h_new);
1002 } else {
1003 return this->irk_step(x_old, t_old, h_old, x_new, h_new);
1004 }
1005
1006 #undef CMD
1007 }

1022 bool advance(VectorN const &x_old, Real t_old, Real h_old, VectorN &x_new, Real &h_new)
1023 {
1024 #define CMD "Sandals::RungeKutta::advance(...): "
1025
1026 // Check step size
1027 SANDALS_ASSERT(h_old > Real(0.0), CMD "in " << this->m_tableau.name << " solver, h = "<<
1028 h_old << ", expected > 0.");
1029
1030 // If the integration step failed, try again with substepping
1031 if (!this->step(x_old, t_old, h_old, x_new, h_new))
1032 {
1033 VectorN x_tmp(x_old);
1034 Real t_tmp{t_old}, h_tmp{h_old / Real(2.0)};
1035
1036 // Substepping logic
1037 Size max_k{this->m_max_substeps * this->m_max_substeps}, k{2};
1038 Real h_new_tmp;
1039 while (k > 0) {
1040 // Calculate the next step with substepping logic
1041 if (this->step(x_tmp, t_tmp, h_tmp, x_new, h_new_tmp)) {
1042
1043 // Accept the step
1044 h_tmp = h_new_tmp;
1045
1046 // If substepping is enabled, double the step size
1047 if (k > 0 && k < max_k) {
1048 k -= 1;
1049 // If the substepping index is even, double the step size
1050 if (k % 2 == 0) {
1051 h_tmp = Real(2.0) * h_tmp;
1052 if (this->m_verbose) {
1053 SANDALS_WARNING(CMD "in " << this->m_tableau.name << " solver, at t = " << t_tmp <<
1054 ", integration succedded disable one substepping layer.");
1055 }
1056 }
1057 }
1058
1059 // Check the infinity norm of the solution
1060 SANDALS_ASSERT(std::isfinite(x_tmp.maxCoeff()), CMD "in " << this->m_tableau.name <<
1061 " solver, at t = " << t_tmp << ", ||x||_inf = inf, computation interrupted.");
1062
1063 } else {
1064
1065 // If the substepping index is too high, abort the integration
1066 k += 2;
1067 SANDALS_ASSERT(k < max_k, CMD "in " << this->m_tableau.name << " solver, at t = " <<
1068 t_tmp << ", integration failed with h = " << h_tmp << ", aborting.");
1069 return false;
1070
1071 // Otherwise, try again with a smaller step
1072 if (this->m_verbose) {SANDALS_WARNING(CMD "in " << this->m_tableau.name << " solver, " <<
1073 "at t = " << t_tmp << ", integration failed, adding substepping layer.");}
1074 h_tmp /= Real(2.0);
1075 continue;
1076 }
1077
1078 // Store independent variable (or time)
1079 t_tmp += h_tmp;
1080 }
1081
1082 // Store output states substepping solutions
1083 x_new = x_tmp;
1084 h_new = h_tmp;
1085 }
1086
1087 // Project intermediate solution on the invariants
1088 if (this->m_projection) {
1089 VectorN x_projected;
1090 if (this->project(x_new, t_old + h_new, x_projected)) {
1091 x_new = x_projected;
1092 return true;
1093 } else {
1094 return false;
1095 }
1096 } else {
1097 return true;
1098 }
1099
1100 #undef CMD
1101 }

1112 bool solve(VectorX const &t_mesh, VectorN const &ics, Solution &sol)
1113 {
1114 using Eigen::last;
1115
1116 // Instantiate output
1117 sol.resize(t_mesh.size());
1118
1119 // Store first step
1120 sol.t(0) = t_mesh(0);
1121 sol.x.col(0) = ics;
1122 sol.h.col(0) = this->m_system->h(ics, t_mesh(0));
1123
1124 // Update the current step
1125 Size step{0};
1126 VectorN x_step{ics};
1127 Real t_step{t_mesh(0)}, h_step{t_mesh(1) - t_mesh(0)}, h_tmp_step{h_step}, h_new_step;
1128 bool mesh_point_bool, saturation_bool;
1129
1130 while (true) {
1131 // Integrate system
1132 if (!this->advance(sol.x.col(step), t_step, h_step, x_step, h_new_step)) {return false;}
1133
1134 // Update the current step
1135 t_step += h_step;
1136
1137 // Saturate the suggested timestep
1138 mesh_point_bool = std::abs(t_step - t_mesh(step+1)) < SQRT_EPSILON;
1139 saturation_bool = t_step + h_new_step > t_mesh(step+1) + SQRT_EPSILON;
1140 if (this->m_adaptive && this->m_tableau.is_embedded && !mesh_point_bool && saturation_bool) {
1141 h_tmp_step = h_new_step; // Used to store the previous step and keep the integration pace
1142 h_step = t_mesh(step+1) - t_step;
1143 } else {
1144 h_step = h_new_step;
1145 }
1146
1147 // Store solution if the step is a mesh point
1148 if (!this->m_adaptive || mesh_point_bool) {
1149 // Update temporaries
1150 step += 1;
1151 h_step = h_tmp_step;
1152
1153 // Update outputs
1154 sol.t(step) = t_step;
1155 sol.x.col(step) = x_step;
1156 sol.h.col(step) = this->m_system->h(x_step, t_step);
1157
1158 // Check if the current step is the last one
1159 if (std::abs(t_step - t_mesh(last)) < SQRT_EPSILON) {break;}
1160 }
1161 }
1162 return true;
1163 }

1174 bool adaptive_solve(VectorX const &t_mesh, VectorN const &ics, Solution &sol)
1175 {
1176 using Eigen::all;
1177 using Eigen::last;
1178
1179 #define CMD "Sandals::RungeKutta::adaptive_solve(...): "
1180
1181 // Check if the adaptive method is enabled and the method is embedded
1182 if (!this->is_embedded()) {
1183 SANDALS_WARNING(CMD "the method is not embedded, using solve(...) method.");
1184 return this->solve(t_mesh, ics, sol);
1185 } else if (!this->m_adaptive) {
1186 SANDALS_WARNING(CMD "adaptive method is disabled, using solve(...) method.");
1187 return this->solve(t_mesh, ics, sol);
1188 }
1189
1190 // Instantiate output
1191 Real h_step{t_mesh(1) - t_mesh(0)}, h_new_step, scale{100.0};
1192 Real h_min{std::max(this->m_min_step, h_step/scale)}, h_max{scale*h_step};
1193 if (this->m_tableau.is_embedded) {
1194 Size safety_length{static_cast<Size>(std::ceil(std::abs(t_mesh(last) - t_mesh(0))/h_min))};
1195 sol.resize(safety_length);
1196 } else {
1197 sol.resize(t_mesh.size());
1198 }
1199
1200 // Store first step
1201 sol.t(0) = t_mesh(0);
1202 sol.x.col(0) = ics;
1203 sol.h.col(0) = this->m_system->h(ics, t_mesh(0));
1204
1205 // Instantiate temporary variables
1206 Size step{0};
1207 VectorN x_step{ics};
1208
1209 while (true) {
1210 // Integrate system
1211 this->advance(sol.x.col(step), sol.t(step), h_step, x_step, h_new_step);
1212
1213 // Saturate the suggested timestep
1214 if (this->m_adaptive && this->m_tableau.is_embedded) {
1215 h_step = std::max(std::min(h_new_step, h_max), h_min);
1216 }
1217
1218 SANDALS_ASSERT(step < t_mesh.size(), CMD "safety length exceeded.");
1219
1220 // Store solution
1221 sol.t(step+1) = sol.t(step) + h_step;
1222 sol.x.col(step+1) = x_step;
1223 sol.h.col(step+1) = this->m_system->h(x_step, sol.t(step+1));
1224
1225 // Check if the current step is the last one
1226 if (sol.t(step+1) + h_step > t_mesh(last)) {break;}
1227
1228 // Update steps counter
1229 step += 1;
1230 }
1231
1232 // Resize the output
1233 sol.conservative_resize(step-1);
1234
1235 return true;
1236
1237 #undef CMD
1238 }

1249 bool project(VectorN const &x, Real t, VectorN &x_projected)
1250 {
1251 #define CMD "Sandals::RungeKutta::project(...): "
1252
1253 // Check if there are any constraints
1254 x_projected = x;
1255 if (M > Size(0)) {
1256
1257 VectorM h;
1258 MatrixM Jh_x;
1259 VectorP b, x_step;
1260 MatrixP A;
1261 A.setZero();
1262 A.template block<N, N>(0, 0) = MatrixN::Identity();
1263 for (Size k{0}; k < this->m_max_projection_iterations; ++k) {
1264
1265 // Standard projection method
1266 // [A] {x} = {b}
1267 // / I Jh_x^T \ / dx \ / x_t - x_k \
1268 // | | | | = | |
1269 // \ Jh_x 0 / \ lambda / \ -h /
1270
1271 // Evaluate the invariants vector and its Jacobian
1272 h = this->m_system->h(x_projected, t);
1273 Jh_x = this->m_system->Jh_x(x_projected, t);
1274
1275 // Check if the solution is found
1276 if (h.norm() < this->m_projection_tolerance) {return true;}
1277
1278 // Build the linear system
1279 A.template block<N, M>(0, N) = Jh_x.transpose();
1280 A.template block<M, N>(N, 0) = Jh_x;
1281 b.template head<N>() = x - x_projected;
1282 b.template tail<M>() = -h;
1283
1284 // Compute the solution of the linear system
1285 this->m_lu.compute(A);
1286 SANDALS_ASSERT(this->m_lu.rank() == N+M, CMD "singular Jacobian detected.");
1287 x_step = this->m_lu.solve(b);
1288
1289 // Check if the step is too small
1290 if (x_step.norm() < this->m_projection_tolerance * this->m_projection_tolerance) {return false;}
1291
1292 // Update the solution
1293 x_projected.noalias() += x_step(Eigen::seqN(0, N));
1294 }
1295 if (this->m_verbose) {SANDALS_WARNING(CMD "maximum number of iterations reached.");}
1296 return false;
1297 } else {
1298 return true;
1299 }
1300
1301 #undef CMD
1302 }

1315 bool project_ics(VectorN const &x, Real t, std::vector<Size> const & projected_equations,
1316 std::vector<Size> const & projected_invariants, VectorN &x_projected) const
1317 {
1318 #define CMD "Sandals::RungeKutta::project_ics(...): "
1319
1320 Size X{static_cast<Size>(projected_equations.size())};
1321 Size H{static_cast<Size>(projected_invariants.size())};
1322
1323 // Check if there are any constraints
1324 x_projected = x;
1325 if (H > Size(0)) {
1326
1327 VectorM h;
1328 MatrixM Jh_x;
1329 VectorX b(X+H), x_step(X+H);
1330 MatrixX A(X+H, X+H);
1331 A.setZero();
1332 A.block(0, 0, X, X) = MatrixX::Identity(X+H, X+H);
1333 Eigen::FullPivLU<MatrixX> lu;
1334 for (Size k{0}; k < this->m_max_projection_iterations; ++k) {
1335
1336 // Standard projection method
1337 // [A] {x} = {b}
1338 // / I Jh_x^T \ / dx \ / x_t - x_k \
1339 // | | | | = | |
1340 // \ Jh_x 0 / \ lambda / \ -h /
1341
1342 // Evaluate the invariants vector and its Jacobian
1343 h = this->m_system->h(x_projected, t);
1344 Jh_x = this->m_system->Jh_x(x_projected, t);
1345
1346 // Select only the projected invariants
1347 h = h(projected_invariants);
1348 Jh_x = Jh_x(projected_invariants, projected_equations);
1349
1350 // Check if the solution is found
1351 if (h.norm() < this->m_projection_tolerance) {return true;}
1352
1353 // Build the linear system
1354 A.block(0, X, X, H) = Jh_x.transpose();
1355 A.block(X, 0, H, X) = Jh_x;
1356 b.head(X) = x(projected_equations) - x_projected(projected_equations);
1357 b.tail(H) = -h;
1358
1359 // Compute the solution of the linear system
1360 lu.compute(A);
1361 SANDALS_ASSERT(lu.rank() == X+H, CMD "singular Jacobian detected.");
1362 x_step = this->m_lu.solve(b);
1363
1364 // Check if the step is too small
1365 if (x_step.norm() < this->m_projection_tolerance * this->m_projection_tolerance) {return false;}
1366
1367 // Update the solution
1368 x_projected(projected_equations).noalias() += x_step;
1369 }
1370 if (this->m_verbose) {SANDALS_WARNING(CMD "maximum number of iterations reached.");}
1371 return false;
1372 } else {
1373 return true;
1374 }

1376 #undef CMD

1386 Real estimate_order(std::vector<VectorX> const &t_mesh, VectorN const &ics, std::function<MatrixX(VectorX)> &sol)
1387 {
1388 using Eigen::last;
1389
1390 #define CMD "Sandals::RungeKutta::estimate_order(...): "
1391
1392 SANDALS_ASSERT(t_mesh.size() > Size(1), CMD "expected at least two time meshes.");
1393
1394 for (Size i{0}; i < static_cast<Size>(t_mesh.size()); ++i) {
1395
1396 // Check if the time mesh has the same initial and final time
1397 SANDALS_ASSERT((t_mesh[0](0) - t_mesh[i](0)) < SQRT_EPSILON,
1398 CMD "expected the same initial time.");
1399 SANDALS_ASSERT((t_mesh[0](last) - t_mesh[i](last)) < SQRT_EPSILON,
1400 CMD "expected the same final time.");
1401
1402 // Check if the mesh step is fixed all over the time mesh
1403 for (Size j{1}; j < static_cast<Size>(t_mesh[i].size()); ++j) {
1404 SANDALS_ASSERT((t_mesh[i](j) - t_mesh[i](j-1)) - (t_mesh[i](1) - t_mesh[i](0)) < SQRT_EPSILON,
1405 CMD "expected a fixed step.");
1406 }
1407 }
1408
1409 // Solve the system for each time scale
1410 Solution sol_num;
1411 MatrixX sol_ana;
1412 VectorX h_vec(t_mesh.size()), e_vec(t_mesh.size());
1413 for (Size i{0}; i < static_cast<Size>(t_mesh.size()); ++i) {
1414 SANDALS_ASSERT(this->solve(t_mesh[i], ics, sol_num), CMD "failed to solve the system for " <<
1415 "the" << i << "-th time mesh.");
1416 sol_ana = sol(sol_num.t);
1417 SANDALS_ASSERT(sol_ana.rows() == sol_num.x.rows(),
1418 CMD "expected the same number of states in analytical solution.");
1419 SANDALS_ASSERT(sol_ana.cols() == sol_num.x.cols(),
1420 CMD "expected the same number of steps in analytical solution.");
1421 h_vec(i) = std::abs(sol_num.t(1) - sol_num.t(0));
1422 e_vec(i) = (sol_ana - sol_num.x).array().abs().maxCoeff();
1423 }
1424
1425 // Compute the order of the method thorugh least squares
1426 VectorX A(h_vec.array().log());
1427 VectorX b(e_vec.array().log());
1428 return ((A.transpose() * A).ldlt().solve(A.transpose() * b))(0);
1429
1430 #undef CMD
1431 }

646 Real estimate_step(VectorN const &x, VectorN const &x_e, Real h_k) const
647 {
648 Real desired_error{this->m_absolute_tolerance + this->m_relative_tolerance *
649 std::max(x.array().abs().maxCoeff(), x_e.array().abs().maxCoeff())};
650 Real truncation_error{(x - x_e).array().abs().maxCoeff()};
651 return h_k * std::min(this->m_max_safety_factor, std::max(this->m_min_safety_factor,
652 this->m_safety_factor * std::pow(desired_error/truncation_error,
653 1.0/std::max(this->m_tableau.order, this->m_tableau.order_e))));
654 }

711 bool erk_explicit_step(VectorN const &x_old, Real t_old, Real h_old, VectorN &x_new, Real &h_new) const
712 {
713 using Eigen::all;
714 using Eigen::seqN;
715
716 // Compute the K variables in the case of an explicit method and explicit system
717 VectorN x_node;
718 MatrixK K;
719 for (Integer i{0}; i < S; ++i) {
720 x_node = x_old + K(all, seqN(0, i)) * this->m_tableau.A(i, seqN(0, i)).transpose();
721 if (!this->m_reverse) {
722 K.col(i) = h_old * static_cast<Explicit<Real, N, M> const *>(this->m_system.get())->f(x_node, t_old + h_old*this->m_tableau.c(i));
723 } else {
724 K.col(i) = h_old * static_cast<Explicit<Real, N, M> const *>(this->m_system.get())->f_reverse(x_node, t_old + h_old*this->m_tableau.c(i));
725 }
726 }
727 if (!K.allFinite()) {return false;}
728
729 // Perform the step and obtain the next state
730 x_new = x_old + K * this->m_tableau.b;
731
732 // Adapt next step
733 if (this->m_adaptive && this->m_tableau.is_embedded) {
734 VectorN x_emb = x_old + K * this->m_tableau.b_e;
735 h_new = this->estimate_step(x_new, x_emb, h_old);
736 }
737 return true;
738 }

899 if (!this->m_reverse) {
900 fun_mat.col(i) = this->m_system->F(x_node, K_mat.col(i)/h, t + h * this->m_tableau.c(i));

944 void irk_jacobian(VectorN const &x, Real t, Real h, VectorK const &K, MatrixJ &jac) const
945 {
946 using Eigen::seqN;
947
948 // Reset the Jacobian matrix
949 jac.setZero();
950
951 // Loop through each equation of the system
952 MatrixK K_mat{K.reshaped(N, S)};
953 Real t_node;
954 VectorN x_node, x_dot_node;
955 MatrixN JF_x, JF_x_dot;
956 auto idx = seqN(0, N), jdx = seqN(0, N);
957 for (Integer i{0}; i < S; ++i) {
958 t_node = t + h * this->m_tableau.c(i);
959 x_node = x + K_mat * this->m_tableau.A.row(i).transpose();
960
961 // Compute the Jacobians with respect to x and x_dot
962 x_dot_node = K_mat.col(i) / h;
963 if (!this->m_reverse) {
964 JF_x = this->m_system->JF_x(x_node, x_dot_node, t_node);
965 JF_x_dot = this->m_system->JF_x_dot(x_node, x_dot_node, t_node);
966 } else {
967 JF_x = this->m_system->JF_x_reverse(x_node, x_dot_node, t_node);
968 JF_x_dot = this->m_system->JF_x_dot_reverse(x_node, x_dot_node, t_node);
969 }
970
971 // Combine the Jacobians with respect to x and x_dot to obtain the Jacobian with respect to K
972 idx = seqN(i*N, N);
973 for (Integer j{0}; j < S; ++j) {
974 jdx = seqN(j*N, N);
975 if (i == j) {
976 jac(idx, jdx) = this->m_tableau.A(i,j) * JF_x + JF_x_dot / h;
977 } else {
978 jac(idx, jdx) = this->m_tableau.A(i,j) * JF_x;
979 }
980 }
981 }
982 }

997 bool irk_step(VectorN const &x_old, Real t_old, Real h_old, VectorN &x_new, Real &h_new)
998 {
999 VectorK K;
1000 VectorK K_ini(VectorK::Zero());
1001
1002 // Check if the solver converged
1003 if (!this->m_newtonK.solve(
1004 [this, &x_old, t_old, h_old](VectorK const &K_fun, VectorK &fun)
1005 {this->irk_function(x_old, t_old, h_old, K_fun, fun);},
1006 [this, &x_old, t_old, h_old](VectorK const &K_jac, MatrixJ &jac)
1007 {this->irk_jacobian(x_old, t_old, h_old, K_jac, jac);},
1008 K_ini, K))
1009 {return false;}
1010
1011 // Perform the step and obtain the next state
1012 x_new = x_old + K.reshaped(N, S) * this->m_tableau.b;
1013

1200 bool advance(VectorN const &x_old, Real t_old, Real h_old, VectorN &x_new, Real &h_new)
1201 {
1202 #define CMD "Sandals::RungeKutta::advance(...): "
1203
1204 // Check step size
1205 SANDALS_ASSERT(h_old > Real(0.0), CMD "in " << this->m_tableau.name << " solver, h = "<<
1206 h_old << ", expected > 0.");
1207
1208 // If the integration step failed, try again with substepping
1209 if (!this->step(x_old, t_old, h_old, x_new, h_new))
1210 {
1211 VectorN x_tmp(x_old);
1212 Real t_tmp{t_old}, h_tmp{h_old / Real(2.0)};
1213
1214 // Substepping logic
1215 Integer max_k{this->m_max_substeps * this->m_max_substeps}, k{2};
1216 Real h_new_tmp;
1217 while (k > 0) {
1218 // Calculate the next step with substepping logic
1219 if (this->step(x_tmp, t_tmp, h_tmp, x_new, h_new_tmp)) {
1220
1221 // Accept the step
1222 h_tmp = h_new_tmp;
1223
1224 // If substepping is enabled, double the step size
1225 if (k > 0 && k < max_k) {
1226 k -= 1;
1227 // If the substepping index is even, double the step size
1228 if (k % 2 == 0) {
1229 h_tmp = Real(2.0) * h_tmp;
1230 if (this->m_verbose) {
1231 SANDALS_WARNING(CMD "in " << this->m_tableau.name << " solver, at t = " << t_tmp <<
1232 ", integration succedded disable one substepping layer.");
1233 }
1234 }
1235 }
1236
1237 // Check the infinity norm of the solution
1238 SANDALS_ASSERT(std::isfinite(x_tmp.maxCoeff()), CMD "in " << this->m_tableau.name <<
1239 " solver, at t = " << t_tmp << ", ||x||_inf = inf, computation interrupted.");
1240
1241 } else {
1242
1243 // If the substepping index is too high, abort the integration
1244 k += 2;
1245 SANDALS_ASSERT(k < max_k, CMD "in " << this->m_tableau.name << " solver, at t = " <<
1246 t_tmp << ", integration failed with h = " << h_tmp << ", aborting.");
1247 return false;
1248
1249 // Otherwise, try again with a smaller step
1250 if (this->m_verbose) {SANDALS_WARNING(CMD "in " << this->m_tableau.name << " solver, " <<
1251 "at t = " << t_tmp << ", integration failed, adding substepping layer.");}
1252 h_tmp /= Real(2.0);
1253 continue;
1254 }
1255
1256 // Store independent variable (or time)
1257 t_tmp += h_tmp;
1258 }
1259
1260 // Store output states substepping solutions
1261 x_new = x_tmp;
1262 h_new = h_tmp;
1263 }
1264
1265 // Project intermediate solution on the invariants
1266 if (this->m_projection) {
1267 VectorN x_projected;
1268 if (this->project(x_new, t_old + h_new, x_projected)) {
1269 x_new = x_projected;
1270 } else {
1271 return false;
1272 }
1273 }
1274
1275 // Callback after the step is completed
1276 if (this->m_step_callback) {
1277 this->m_step_callback(x_new, t_old + h_new);
1278 }
1279 return true;

1295 bool solve(VectorX const &t_mesh, VectorN const &ics, Solution<Real, N, M> &sol)
1296 {
1297 using Eigen::last;
1298
1299 // Instantiate output
1300 sol.resize(t_mesh.size());
1301
1302 // Store first step
1303 sol.t(0) = t_mesh(0);
1304 sol.x.col(0) = ics;
1305 sol.h.col(0) = this->m_system->h(ics, t_mesh(0));
1306
1307 // Update the current step
1308 Integer step{0};
1309 VectorN x_step{ics};
1310 Real t_step{t_mesh(0)}, h_step{t_mesh(1) - t_mesh(0)}, h_tmp_step{h_step}, h_new_step;
1311 bool mesh_point_bool, saturation_bool;
1312
1313 while (true) {
1314 // Integrate system
1315 if (!this->advance(sol.x.col(step), t_step, h_step, x_step, h_new_step)) {return false;}
1316
1317 // Update the current step
1318 t_step += h_step;
1319
1320 // Saturate the suggested timestep
1321 mesh_point_bool = std::abs(t_step - t_mesh(step+1)) < SQRT_EPSILON;
1322 saturation_bool = t_step + h_new_step > t_mesh(step+1) + SQRT_EPSILON;
1323 if (this->m_adaptive && this->m_tableau.is_embedded && !mesh_point_bool && saturation_bool) {
1324 h_tmp_step = h_new_step; // Used to store the previous step and keep the integration pace
1325 h_step = t_mesh(step+1) - t_step;
1326 } else {
1327 h_step = h_new_step;
1328 }
1329
1330 // Store solution if the step is a mesh point
1331 if (!this->m_adaptive || mesh_point_bool) {
1332 // Update temporaries
1333 step += 1;
1334 h_step = h_tmp_step;
1335
1336 // Update outputs
1337 sol.t(step) = t_step;
1338 sol.x.col(step) = x_step;
1339 sol.h.col(step) = this->m_system->h(x_step, t_step);
1340
1341 // Check if the current step is the last one
1342 if (std::abs(t_step - t_mesh(last)) < SQRT_EPSILON) {break;}
1343 }
1344 }
1345 return true;

1359 bool adaptive_solve(VectorX const &t_mesh, VectorN const &ics, Solution<Real, N, M> &sol)
1360 {

1363
1364 #define CMD "Sandals::RungeKutta::adaptive_solve(...): "
1365
1366 // Check if the adaptive method is enabled and the method is embedded
1367 if (!this->is_embedded()) {
1368 SANDALS_WARNING(CMD "the method is not embedded, using solve(...) method.");
1369 return this->solve(t_mesh, ics, sol);
1370 } else if (!this->m_adaptive) {
1371 SANDALS_WARNING(CMD "adaptive method is disabled, using solve(...) method.");
1372 return this->solve(t_mesh, ics, sol);
1373 }
1374
1375 // Instantiate output
1376 Real h_step{t_mesh(1) - t_mesh(0)}, h_new_step, scale{100.0};
1377 Real h_min{std::max(this->m_min_step, h_step/scale)}, h_max{scale*h_step};
1378 if (this->m_tableau.is_embedded) {
1379 Integer safety_length{static_cast<Integer>(std::ceil(std::abs(t_mesh(last) - t_mesh(0))/(2.0*h_min)))};
1380 sol.resize(safety_length);
1381 } else {
1382 sol.resize(t_mesh.size());
1383 }
1384
1385 // Store first step
1386 sol.t(0) = t_mesh(0);
1387 sol.x.col(0) = ics;

1389

1391 Integer step{0};
1392 VectorN x_step{ics};
1393
1394 while (true) {
1395 // Integrate system
1396 this->advance(sol.x.col(step), sol.t(step), h_step, x_step, h_new_step);
1397
1398 // Saturate the suggested timestep
1399 if (this->m_adaptive && this->m_tableau.is_embedded) {
1400 h_step = std::max(std::min(h_new_step, h_max), h_min);
1401 }
1402
1403 SANDALS_ASSERT(step < sol.size(), CMD "safety length exceeded.");
1404
1405 // Store solution
1406 sol.t(step+1) = sol.t(step) + h_step;
1407 sol.x.col(step+1) = x_step;
1408 sol.h.col(step+1) = this->m_system->h(x_step, sol.t(step+1));
1409
1410 // Check if the current step is the last one
1411 if (sol.t(step+1) + h_step > t_mesh(last)) {break;}
1412
1413 // Update steps counter
1414 step += 1;
1415 }
1416
1417 // Resize the output
1418 sol.conservative_resize(step-1);
1419
1420 return true;

1434 bool project(VectorN const &x, Real t, VectorN &x_projected)
1435 {
1436 #define CMD "Sandals::RungeKutta::project(...): "
1437
1438 // Check if there are any constraints
1439 x_projected = x;
1440 if (M > Integer(0)) {
1441
1442 VectorM h;
1443 MatrixM Jh_x;
1444 VectorP b, x_step;
1445 MatrixP A;
1446 A.setZero();
1447 A.template block<N, N>(0, 0) = MatrixN::Identity();
1448 for (Integer k{0}; k < this->m_max_projection_iterations; ++k) {
1449
1450 /* Standard projection method
1451 [A] {x} = {b}
1452 / I Jh_x^T \ / dx \ / x_t - x_k \
1453 | | | | = | |
1454 \ Jh_x 0 / \ lambda / \ -h /
1455 */
1456
1457 // Evaluate the invariants vector and its Jacobian
1458 h = this->m_system->h(x_projected, t);
1459 Jh_x = this->m_system->Jh_x(x_projected, t);
1460
1461 // Check if the solution is found
1462 if (h.norm() < this->m_projection_tolerance) {return true;}
1463
1464 // Build the linear system
1465 A.template block<N, M>(0, N) = Jh_x.transpose();
1466 A.template block<M, N>(N, 0) = Jh_x;
1467 b.template head<N>() = x - x_projected;
1468 b.template tail<M>() = -h;
1469
1470 // Compute the solution of the linear system
1471 this->m_lu.compute(A);
1472 SANDALS_ASSERT(this->m_lu.rank() == N+M, CMD "singular Jacobian detected.");
1473 x_step = this->m_lu.solve(b);
1474
1475 // Check if the step is too small
1476 if (x_step.norm() < this->m_projection_tolerance * this->m_projection_tolerance) {return false;}
1477
1478 // Update the solution
1479 x_projected.noalias() += x_step(Eigen::seqN(0, N));
1480 }
1481 if (this->m_verbose) {SANDALS_WARNING(CMD "maximum number of iterations reached.");}
1482 return false;
1483 } else {
1484 return true;

445 real estimate_step(vecN const &x_h, vecN const &x_l, real d_t) const
446 {

449 this->m_abs_tol + this->m_rel_tol*std::max(x_h.cwiseAbs().maxCoeff(), x_l.cwiseAbs().maxCoeff())
450 )).cwiseAbs().maxCoeff()));

558 bool explicit_step(vecN const &x_k, real t_k, real t_step, vecN &x_out, real &t_adap) const
559 {
560 using Eigen::all;
561 using Eigen::seqN;
562
563 // Compute the K variables in the case of an explicit method and explicit system
564 real t_i;
565 vecN x_i;
566 matK K;
567 for (unsigned i = 0; i < S; ++i)
568 {
569 t_i = t_k + this->m_tableau.c(i) * t_step;
570 x_i = x_k + K(all, seqN(0, i)) * this->m_tableau.A(i, seqN(0, i)).transpose();
571 K.col(i) = t_step * static_cast<Explicit<N> const *>(this->m_system.get())->f(x_i, t_i);
572 }
573 if (!K.allFinite()) {return false;}
574
575 // Perform the step and obtain the next state
576 x_out = x_k + K * this->m_tableau.b;
577
578 // Adapt next time step
579 if (this->m_adaptive_step && this->m_is_embedded) {
580 vecN x_e = x_k + K * this->m_tableau.b_e;
581 t_adap = this->estimate_step(x_out, x_e, t_step);
582 }
583 return true;
584 }

599 void implicit_function(vecN const &x_k, vecK const &K_k, real t_k, real d_t, vecK &fun) const
600 {
601 fun.setZero();
602
603 // Loop through each equation of the system
604 real t_i;
605 vecN x_i;
606 matK K{K_k.reshaped(N, S)}; // FIXME: can i avoid reshaping?
607 matK fun_tmp;
608 for (unsigned i = 0; i < S; ++i) {
609 t_i = t_k + this->m_tableau.c(i) * d_t;
610 x_i = x_k + K * this->m_tableau.A.row(i).transpose();
611 fun_tmp.col(i) = this->m_system->F(x_i, K.col(i) / d_t, t_i);
612 }
613 fun = fun_tmp.reshaped(N*S, 1);
614 }

638 void implicit_jacobian(vecN const &x_k, vecK const &K_k, real t_k, real d_t, matJ &jac) const
639 {
640 using Eigen::all;
641 using Eigen::seqN;
642
643 jac.setZero();
644
645 // Loop through each equation of the system
646 matK K{K_k.reshaped(N, S)};
647 real t_i;
648 vecN x_i, x_dot_i;
649 matN JF_x, JF_x_dot;
650 auto idx = seqN(0, N), jdx = seqN(0, N);
651 for (unsigned i = 0; i < S; ++i) {
652 t_i = t_k + this->m_tableau.c(i) * d_t;
653 x_i = x_k + K * this->m_tableau.A.row(i).transpose();
654
655 // Compute the Jacobians with respect to x and x_dot
656 x_dot_i = K.col(i) / d_t;
657 JF_x = this->m_system->JF_x(x_i, x_dot_i, t_i);
658 JF_x_dot = this->m_system->JF_x_dot(x_i, x_dot_i, t_i);
659
660 // Derivative of F(x_i, K(:,i)/d_t, t_i)
661 idx = seqN(i*N, N);
662 for (unsigned j = 0; j < S; ++j) {
663 // Combine the Jacobians with respect to x and x_dot to obtain the
664 // Jacobian with respect to K
665 jdx = seqN(j*N, N);
666 if (i == j) {
667 jac(idx, jdx) = this->m_tableau.A(i,j) * JF_x + JF_x_dot / d_t;
668 } else {
669 jac(idx, jdx) = this->m_tableau.A(i,j) * JF_x;
670 }
671 }
672 }

745 bool implicit_step(vecN const &x_k, real t_k, real d_t, vecN &x_o, real &d_t_star)
746 {
747 // Check if the solver converged
748 vecK K;
749
750 if (!this->m_newton.solve(
751 [this, &x_k, t_k, d_t](vecK const &K, vecK &fun) {this->implicit_function(x_k, K, t_k, d_t, fun);},
752 [this, &x_k, t_k, d_t](vecK const &K, matJ &jac) {this->implicit_jacobian(x_k, K, t_k, d_t, jac);},
753 vecK::Zero(), K)) {return false;}
754
755 // Perform the step and obtain x_k+1
756 x_o = x_k + K.reshaped(N, S) * this->m_tableau.b;
757
758 // Adapt next time step
759 if (this->m_adaptive_step && this->m_tableau.is_embedded) {
760 vecN x_e{x_k + K.reshaped(N, S) * this->m_tableau.b_e};
761 d_t_star = this->estimate_step(x_o, x_e, d_t);
762 }
763 return true;
764 }

816 bool advance(vecN const &x_k, real t_k, real d_t, vecN &x_new, real &d_t_star)
817 {
818 #define CMD "Indigo.RungeKutta.advance(...): "
819
820 // Check step size
821 SANDALS_ASSERT(d_t > real(0.0), CMD "in " << this->m_tableau.name <<
822 "solver, d_t = "<< d_t << ", expected > 0.");
823
824 // If the integration step failed, try again with substepping
825 if (!this->step(x_k, t_k, d_t, x_new, d_t_star))
826 {
827 vecN x_tmp = x_k;
828 real t_tmp = t_k;
829 real d_t_tmp = d_t / real(2.0);
830
831 // Substepping logic
832 unsigned max_substeps = 10;
833 unsigned max_k = max_substeps * max_substeps;
834 unsigned k = 2;
835 real d_t_star_tmp;
836 while (k > 0)
837 {
838 // Calculate the next time step with substepping logic
839 if (this->step(x_tmp, t_tmp, d_t_tmp, x_new, d_t_star_tmp)) {
840
841 // Accept the step
842 d_t_tmp = d_t_star_tmp;
843
844 // If substepping is enabled, double the step size
845 if (k > 0 && k < max_k) {
846 k = k - 1;
847 // If the substepping index is even, double the step size
848 if (k % 2 == 0) {
849 d_t_tmp = real(2.0) * d_t_tmp;
850 if (this->m_verbose) {
851 SANDALS_WARNING(CMD "in " << this->m_tableau.name << " solver, at t = " <<
852 t_tmp << ", integration succedded disable one substepping layer.");
853 }
854 }
855 }
856
857 // Check the infinity norm of the solution
858 SANDALS_ASSERT(std::isfinite(x_tmp.maxCoeff()), CMD "in " <<
859 this->m_tableau.name << " solver, at t = " << t_tmp <<
860 ", ||x||_inf = inf, computation interrupted.");
861
862 } else {
863
864 // If the substepping index is too high, abort the integration
865 k = k + 2;
866 SANDALS_ASSERT(k < max_k, CMD "in " << this->m_tableau.name <<
867 " solver, at t = " << t_tmp << ", integration failed with d_t = " <<
868 d_t << ", aborting.");
869 return false;
870
871 // Otherwise, try again with a smaller step
872 if (this->m_verbose) {
873 SANDALS_WARNING(CMD "in " << this->m_tableau.name << " solver, at t = " <<
874 t_tmp << ", integration failed, adding substepping layer.");
875 }
876 d_t_tmp = d_t_tmp / real(2.0);
877 continue;
878
879 }
880
881 // Store time solution
882 t_tmp = t_tmp + d_t_tmp;
883 }
884
885 // Store output states substepping solutions
886 x_new = x_tmp;
887 d_t_star = d_t_tmp;
888 return true;
889 } else {
890 return true;
891 }
892 #undef CMD
893 }

905 bool solve(vecD const &t, vecN const &ics, solution &sol)
906 {
907 using Eigen::last;
908
909 // Instantiate output
910 sol.resize(t.size());
911
912 // Store first step
913 sol.time(0) = t(0);
914 sol.states.col(0) = ics;
915
916 // Update the current step
917 unsigned s{0};
918 vecN x_s{ics};
919 real t_s{t(0)};
920 real d_t_s{t(1) - t(0)};
921 real d_t_tmp{d_t_s}, d_t_star;
922 bool mesh_point_bool, saturation_bool;
923
924 while (true) {
925 // Integrate system
926 if (!this->advance(sol.states.col(s), t_s, d_t_s, x_s, d_t_star)) {return false;}
927
928 // Update the current step
929 t_s = t_s + d_t_s;
930
931 // Saturate the suggested timestep
932 mesh_point_bool = std::abs(t_s - t(s+1)) < SQRT_EPSILON;
933 saturation_bool = t_s + d_t_star > t(s+1) + SQRT_EPSILON;
934 if (this->m_adaptive_step && !mesh_point_bool && saturation_bool) {
935 d_t_tmp = d_t_star;
936 d_t_s = t(s+1) - t_s;
937 } else {
938 d_t_s = d_t_star;
939 }
940
941 // Store solution if the step is a mesh point
942 if (!this->m_adaptive_step || mesh_point_bool) {
943 // Update temporaries
944 s += 1;
945 d_t_s = d_t_tmp;
946
947 // Update outputs
948 sol.time(s) = t_s;
949 sol.states.col(s) = x_s;
950
951 // Check if the current step is the last one
952 if (std::abs(t_s - t(last)) < SQRT_EPSILON) {break;}
953 }
954 }
955 return true;
956 }

967 bool adaptive_solve(vecD const &t, vecN const &ics, solution &sol)
968 {
969 using Eigen::last;
970
971 #define CMD "Sandals::RungeKutta::adaptive_solve(...): "
972
973 // Instantiate output
974 this->enable_adaptive_step();
975 real d_t{t(1) - t(0)}, d_t_star;
976 if (this->m_tableau.is_embedded) {
977 real scale{100.0};
978 real t_min{std::max(this->m_d_t_min, d_t/scale)};
979 real t_max{scale*d_t};
980 unsigned safety_length = std::ceil(real(1.5)/this->m_min_sft_fac) * t.size();
981 sol.resize(safety_length);
982 } else {
983 sol.resize(t.size());
984 }
985
986 // Store first step
987 sol.time(0) = t(0);
988 sol.states.col(0) = ics;
989
990 // Instantiate temporary variables
991 unsigned s{0};
992 vecN x_s{ics};
993
994 while (true) {
995 // Integrate system
996 this->advance(sol.states.col(s), sol.time(s), d_t, x_s, d_t_star);
997
998 // Saturate the suggested timestep
999 d_t = std::max(std::min(d_t_star, t_max), t_min);
1000 std::cout << "d_t = " << d_t << std::endl;
1001
1002 SANDALS_ASSERT(s < safety_length, CMD "safety length exceeded.");
1003
1004 // Store solution
1005 sol.time(s+1) = sol.time(s) + d_t;
1006 sol.states.col(s+1) = x_s;
1007
1008 // Check if the current step is the last one
1009 if (sol.time(s+1) + d_t > t(last)) {break;}
1010
1011 // Update steps counter
1012 s += 1;
1013 }
1014
1015 // Resize the output
1016 sol.conservative_resize(s-1);
1017
1018 return true;
